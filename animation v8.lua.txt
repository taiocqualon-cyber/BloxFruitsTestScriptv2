-- === USER SETTINGS ===
local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local hrp = char:WaitForChild("HumanoidRootPart")
local cam = workspace.CurrentCamera
local TweenService = game:GetService("TweenService")
local Lighting = game:GetService("Lighting")
local RunService = game:GetService("RunService")

-- === MENU ICON NHỎ GỌN ===
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Parent = game.CoreGui
ScreenGui.ResetOnSpawn = false
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

local iconButton = Instance.new("ImageButton")
iconButton.Parent = ScreenGui
iconButton.Size = UDim2.new(0,40,0,40)
iconButton.Position = UDim2.new(0.05,0,0.05,0)
iconButton.BackgroundColor3 = Color3.fromRGB(255,215,0)
iconButton.Image = "rbxassetid://15059956542" -- đổi icon tùy ý
iconButton.ZIndex = 10

-- Toggle domain cinematic
local DomainActive = false

iconButton.MouseButton1Click:Connect(function()
	DomainActive = not DomainActive
	if DomainActive then
		print("💛 Gold Sukuna Domain ON")
	else
		print("💛 Gold Sukuna Domain OFF")
		StopDomain()
	end
end)

-- === STOP DOMAIN FUNCTION ===
function StopDomain()
	DomainActive = false
	cam.CameraType = Enum.CameraType.Custom
	-- Remove particles & lights
	for _,v in pairs(hrp:GetChildren()) do
		if v:IsA("ParticleEmitter") or v:IsA("PointLight") then
			v:Destroy()
		end
	end
	-- Reset lighting
	Lighting.ColorShift_Top = Color3.fromRGB(0,0,0)
	Lighting.ColorShift_Bottom = Color3.fromRGB(0,0,0)
	Lighting.Ambient = Color3.fromRGB(192,192,192)
	Lighting.OutdoorAmbient = Color3.fromRGB(192,192,192)
end

-- === DOMAIN CINEMATIC FUNCTION ===
function RunDomain()
	DomainActive = true

	-- Hover nhân vật liên tục
	local hoverTween = TweenService:Create(hrp, TweenInfo.new(10, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
		CFrame = hrp.CFrame + Vector3.new(0,2,0)
	})
	hoverTween:Play()

	-- Camera đối diện
	cam.CameraType = Enum.CameraType.Scriptable
	cam.CFrame = CFrame.new(hrp.Position + Vector3.new(0,5,10), hrp.Position)

	-- Aura particle
	local aura = Instance.new("ParticleEmitter", hrp)
	aura.Rate = 80
	aura.Lifetime = NumberRange.new(1,2)
	aura.Speed = NumberRange.new(1,3)
	aura.Size = NumberSequence.new(2)
	aura.Color = ColorSequence.new({
		ColorSequenceKeypoint.new(0, Color3.fromRGB(255,0,0)),
		ColorSequenceKeypoint.new(1, Color3.fromRGB(255,215,0))
	})

	-- Mắt đỏ
	local eyes = Instance.new("PointLight", hrp)
	eyes.Color = Color3.fromRGB(255,0,0)
	eyes.Range = 5
	eyes.Brightness = 2

	-- 4 vòng xoay camera sau dịch chuyển
	task.spawn(function()
		local domainCenter = Vector3.new(1000,100,1000)
		hrp.CFrame = CFrame.new(domainCenter + Vector3.new(0,2,0))

		for lap=1,4 do
			for i=1,120 do
				if not DomainActive then return end
				RunService.RenderStepped:Wait()
				local rad = math.rad(i*3)
				local offset = Vector3.new(math.cos(rad)*15,5,math.sin(rad)*15)
				cam.CFrame = CFrame.new(hrp.Position + offset, hrp.Position)
			end
		end

		-- Cuối cùng đối diện nhân vật
		cam.CFrame = CFrame.new(hrp.Position + Vector3.new(0,5,15), hrp.Position)
	end)

	-- Sét phía sau 3 lần
	task.spawn(function()
		for i=1,3 do
			if not DomainActive then return end
			local bolt = Instance.new("Part", workspace)
			bolt.Size = Vector3.new(1,50,1)
			bolt.Anchored = true
			bolt.Material = Enum.Material.Neon
			bolt.Color = Color3.fromRGB(255,255,255)
			bolt.Position = hrp.Position - hrp.CFrame.LookVector*10 + Vector3.new(0,25,0)
			local flash = Instance.new("PointLight", bolt)
			flash.Color = Color3.fromRGB(255,255,255)
			flash.Range = 20
			flash.Brightness = 5
			task.delay(0.3, function() bolt:Destroy() end)
			task.wait(0.5)
		end
	end)

	-- Hover + particle
	TweenService:Create(hrp, TweenInfo.new(5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
		CFrame = hrp.CFrame + Vector3.new(0,2,0)
	}):Play()
end

-- === CHECK ICON CLICK TO RUN ===
iconButton.MouseButton1Click:Connect(function()
	if DomainActive then
		StopDomain()
	else
		RunDomain()
	end
end)
