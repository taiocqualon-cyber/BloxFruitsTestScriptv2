-- 👹💀 RÂTXÂU GOD SYSTEM FULL

local Players = game:GetService("Players")
local plr = Players.LocalPlayer
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

-- ===== SETTINGS =====
local MODE = "BALANCED" -- LOW / HIGH / BALANCED

-- ===== UI =====
local gui = Instance.new("ScreenGui", plr.PlayerGui)

local icon = Instance.new("TextButton", gui)
icon.Size = UDim2.new(0,50,0,50)
icon.Position = UDim2.new(0,10,0.5,0)
icon.Text = "👹"

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,300,0,350)
frame.Position = UDim2.new(0.5,-150,0.5,-175)
frame.BackgroundColor3 = Color3.fromRGB(10,0,0)
frame.Visible = false
frame.Active = true
frame.Draggable = true

icon.MouseButton1Click:Connect(function()
    frame.Visible = not frame.Visible
end)

-- ===== NOTIFY =====
local function notify(msg)
    print("💀 "..msg)
end

-- ===== TOGGLE =====
local function createToggle(name, y, callback)
    local state = false

    local btn = Instance.new("TextButton", frame)
    btn.Size = UDim2.new(1,0,0,40)
    btn.Position = UDim2.new(0,0,0,y)
    btn.BackgroundColor3 = Color3.fromRGB(30,0,0)
    btn.Text = name.." : OFF"
    btn.TextColor3 = Color3.fromRGB(255,0,0)

    btn.MouseButton1Click:Connect(function()
        state = not state
        btn.Text = name.." : "..(state and "ON" or "OFF")
        btn.BackgroundColor3 = state and Color3.fromRGB(0,80,0) or Color3.fromRGB(30,0,0)
        callback(state)
    end)
end

-- ===== STATES =====
local autoFruit, esp, autoChest = false,false,false

-- ===== ESP FIX =====
local function isFruit(v)
    return v:IsA("Tool") and v:FindFirstChild("Handle") and v.Parent == workspace
end

local function addESP(v)
    if v:FindFirstChild("ESP") then return end
    local b = Instance.new("BillboardGui", v)
    b.Name = "ESP"
    b.Size = UDim2.new(0,120,0,40)
    b.AlwaysOnTop = true

    local t = Instance.new("TextLabel", b)
    t.Size = UDim2.new(1,0,1,0)
    t.BackgroundTransparency = 1
    t.Text = "🍈 "..v.Name
    t.TextColor3 = Color3.fromRGB(255,255,0)
end

local function removeESP()
    for _,v in pairs(workspace:GetDescendants()) do
        if v:FindFirstChild("ESP") then
            v.ESP:Destroy()
        end
    end
end

-- ===== STORE =====
local function store()
    for _,t in pairs(plr.Backpack:GetChildren()) do
        if t:IsA("Tool") then
            pcall(function()
                ReplicatedStorage.Remotes.CommF_:InvokeServer("StoreFruit", t.Name)
            end)
        end
    end
end

-- ===== CHEST FIX =====
local function findChest()
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and string.find(v.Name:lower(),"chest") then
            if v:FindFirstChild("HumanoidRootPart") then
                return v.HumanoidRootPart
            end
        end
    end
end

-- ===== PLAYER CHECK =====
local function playerNear()
    for _,p in pairs(Players:GetPlayers()) do
        if p~=plr and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            if (plr.Character.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude < 40 then
                return true
            end
        end
    end
end

-- ===== AI SERVER =====
local visited = {}

local function rateServer(s)
    local score = 0
    local percent = s.playing / s.maxPlayers

    if MODE == "LOW" and s.playing <= 10 then score += 5 end
    if MODE == "HIGH" and percent > 0.6 then score += 5 end
    if MODE == "BALANCED" and percent > 0.3 and percent < 0.8 then score += 5 end

    if s.playing <= 5 then score -= 3 end
    if s.playing >= s.maxPlayers - 2 then score -= 5 end

    return score + math.random()
end

local function changeServer()
    notify("🧠 AI tìm server...")

    local best,score = nil,-999

    local ok,res = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
    end)

    if ok then
        local data = HttpService:JSONDecode(res)

        for _,s in pairs(data.data) do
            if not visited[s.id] then
                local sc = rateServer(s)
                if sc > score then
                    score = sc
                    best = s
                end
            end
        end
    end

    if best then
        visited[best.id] = true
        notify("🚀 "..best.playing.."/"..best.maxPlayers)
        TeleportService:TeleportToPlaceInstance(game.PlaceId,best.id,plr)
    else
        wait(2)
        changeServer()
    end
end

-- ===== BUTTONS =====
createToggle("Auto Fruit",10,function(v) autoFruit=v end)

createToggle("ESP Fruits",60,function(v)
    esp=v
    if not v then removeESP() end
end)

createToggle("Auto Chest",110,function(v) autoChest=v end)

-- SERVER BUTTON
local hopBtn = Instance.new("TextButton", frame)
hopBtn.Size = UDim2.new(1,0,0,40)
hopBtn.Position = UDim2.new(0,0,0,170)
hopBtn.Text = "🔁 Change Server"
hopBtn.MouseButton1Click:Connect(changeServer)

-- TELEPORT
local tpBtn = Instance.new("TextButton", frame)
tpBtn.Size = UDim2.new(1,0,0,40)
tpBtn.Position = UDim2.new(0,0,0,220)
tpBtn.Text = "🚀 Teleport Fruit"

tpBtn.MouseButton1Click:Connect(function()
    for _,v in pairs(workspace:GetDescendants()) do
        if isFruit(v) then
            plr.Character.HumanoidRootPart.CFrame = v.Handle.CFrame
            break
        end
    end
end)

-- ===== LOOP =====
while true do
    if esp then
        for _,v in pairs(workspace:GetDescendants()) do
            if isFruit(v) then addESP(v) end
        end
    end

    if autoFruit then
        if playerNear() then changeServer() end

        for _,v in pairs(workspace:GetDescendants()) do
            if isFruit(v) then
                plr.Character.HumanoidRootPart.CFrame = v.Handle.CFrame
                wait(0.2)
                store()
                notify("🍈 "..v.Name)
                break
            end
        end
    end

    if autoChest then
        local chest = findChest()
        if chest then
            plr.Character.HumanoidRootPart.CFrame = chest.CFrame
            notify("💰 Chest")
            wait(math.random(1,2))
        end
    end

    wait(1)
end
