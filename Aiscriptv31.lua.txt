-- 1. Silent Aim Hook (99% Hit Rate)
local mt = getrawmetatable(game)
local old = mt.__index
setreadonly(mt, false)
mt.__index = newcclosure(function(t, k)
    if t == mouse and (k == "Hit" or k == "Target") and Config.SilentAim and Target and Target.Character then
        local root = Target.Character:FindFirstChild("HumanoidRootPart")
        if root and Target.Character.Humanoid.Health > 0 then
            local pred = root.CFrame + (root.Velocity * Config.Prediction)
            return (k == "Hit" and pred or root)
        end
    end
    return old(t, k)
end)
setreadonly(mt, true)

-- 2. Smart Target & Loop
task.spawn(function()
    while task.wait(0.1) do
        -- Tìm Target gần nhất
        local near = nil
        local dist = 600
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                local d = (player.Character.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if d < dist then dist = d near = p end
            end
        end
        Target = near

        -- Auto Click
        if Config.AutoClick and Target and (player.Character.HumanoidRootPart.Position - Target.Character.HumanoidRootPart.Position).Magnitude < 40 then
            game:GetService("VirtualUser"):CaptureController()
            game:GetService("VirtualUser"):ClickButton1(Vector2.new(0,0))
        end
    end
end)

-- 3. Visuals & ESP (Optimized)
local function drawESP(p)
    local line = Drawing.new("Line")
    local txt = Drawing.new("Text")
    RunService.RenderStepped:Connect(function()
        if p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p ~= player then
            local pos, onScreen = camera:WorldToViewportPoint(p.Character.HumanoidRootPart.Position)
            
            line.Visible = (Config.ShowTracers and onScreen)
            line.From = Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y)
            line.To = Vector2.new(pos.X, pos.Y)
            line.Color = Color3.new(1, 0, 0)

            txt.Visible = (Config.AutoESP and onScreen)
            txt.Position = Vector2.new(pos.X, pos.Y - 50)
            txt.Center = true
            txt.Outline = true
            txt.Size = 16
            local lv = p.Data:FindFirstChild("Level") and p.Data.Level.Value or "?"
            txt.Text = p.DisplayName .. " [Lv." .. lv .. "]"
            txt.Color = (tonumber(lv) or 0) > player.Data.Level.Value and Color3.new(1,0,0) or Color3.new(0,1,0)
        else
            line.Visible = false
            txt.Visible = false
        end
    end)
end
for _, p in pairs(Players:GetPlayers()) do drawESP(p) end
Players.PlayerAdded:Connect(drawESP)
