-- [[ LOGIC CORE ]] --

function GhostFarmLogic()
    spawn(function()
        while _G.AutoFarm do
            task.wait(0.1)
            pcall(function()
                local player = game.Players.LocalPlayer
                local char = player.Character
                
                -- Check Quest Snow Island
                if not player.PlayerGui.Main.Quest.Visible then
                    char.HumanoidRootPart.CFrame = CFrame.new(1385, 87, 155)
                end

                local tool = player.Backpack:FindFirstChild(_G.Weapon) or char:FindFirstChild(_G.Weapon)
                if tool and tool.Parent ~= char then char.Humanoid:EquipTool(tool) end

                for _, v in pairs(game.Workspace.Enemies:GetChildren()) do
                    if (v.Name == "Snowman" or v.Name == "Snow Bandit") and v.Humanoid.Health > 0 then
                        char.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0, 15, 0)
                        if _G.BringMob then
                            v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, -10, 0)
                            v.HumanoidRootPart.CanCollide = false
                        end
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                        task.wait(0.1)
                    end
                end
            end)
        end
    end)
end

function GhostChestLogic()
    spawn(function()
        while _G.AutoChest do
            task.wait(0.1)
            for _, v in pairs(game.Workspace:GetChildren()) do
                if v.Name:find("Chest") and v:IsA("Part") then
                    local dist = (v.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                    local tween = game:GetService("TweenService"):Create(game.Players.LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(dist/_G.TweenSpeed), {CFrame = v.CFrame})
                    tween:Play()
                    tween.Completed:Wait()
                end
            end
        end
    end)
end

function TeleportToFruit()
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and v.Name:find("Fruit") then
            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.Handle.CFrame
            return
        end
    end
end

-- ESP Sytem

                end
            end
        end
    end
end)

-- [[ HOP SERVER LOGIC ]] --
function HopServer()
    local Http = game:GetService("HttpService")
    local TPS = game:GetService("TeleportService")
    local Api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
    local function ListServers(cursor)
        local Raw = game:HttpGet(Api .. ((cursor and "&cursor=" .. cursor) or ""))
        return Http:JSONDecode(Raw)
    end
    local Servers = ListServers()
    for _, s in pairs(Servers.data) do
        if s.playing < s.maxPlayers and s.id ~= game.JobId then
            TPS:TeleportToPlaceInstance(game.PlaceId, s.id, game.Players.LocalPlayer)
            break
        end
    end
end

function CheckMoonAndHop()
    if game:GetService("Lighting").ClockTime >= 18 or game:GetService("Lighting").ClockTime <= 5 then
        Rayfield:Notify({Title = "Moon Status", Content = "Đang là ban đêm! Đang tìm Full Moon...", Duration = 3})
    end
    HopServer()
end

function HopServerHaki()
    Rayfield:Notify({Title = "Haki Check", Content = "Đang quét server Sea 2...", Duration = 3})
    HopServer()
end

function FruitHop()
    local found = false
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and v.Name:find("Fruit") then found = true; break end
    end
    if found then Rayfield:Notify({Title = "Fruit!", Content = "Có trái! Đừng hop!", Duration = 5})
    else HopServer() end
end
