-- ==========================================================
-- [[ PHẦN 1: AUTO FARM & COMBAT ]]
-- ==========================================================
MainTab:CreateSection("Cày Cấp & Gom Quái")

MainTab:CreateToggle({
   Name = "Bật Auto Farm (Ghost Mode)",
   CurrentValue = false,
   Flag = "AutoFarmToggle",
   Callback = function(Value)
      _G.Settings.AutoFarm = Value
      if Value then
         spawn(function()
            while _G.Settings.AutoFarm do
               wait()
               pcall(function()
                  local char = game.Players.LocalPlayer.Character
                  local root = char.HumanoidRootPart
                  -- Tự nhận Quest (Ví dụ Sea 1)
                  if not game.Players.LocalPlayer.PlayerGui.Main.Quest.Visible then
                     game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", "BanditQuest1", 1)
                  end
                  -- Tìm quái và bay đến
                  for _, v in pairs(game.Workspace.Enemies:GetChildren()) do
                     if v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                        root.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0, _G.Settings.DistanceY, 0)
                        -- Gom quái
                        if _G.Settings.BringMob then
                           v.HumanoidRootPart.CanCollide = false
                           v.HumanoidRootPart.CFrame = root.CFrame * CFrame.new(0, -_G.Settings.DistanceY, 0)
                        end
                        -- Đánh nhanh
                        if _G.Settings.FastAttack then
                           game:GetService("VirtualUser"):CaptureController()
                           game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                        end
                     end
                  end
               end)
            end
         end)
      end
   end,
})

MainTab:CreateToggle({
   Name = "Gom Quái (Bring Mobs)",
   CurrentValue = false,
   Flag = "BringToggle",
   Callback = function(Value) _G.Settings.BringMob = Value end,
