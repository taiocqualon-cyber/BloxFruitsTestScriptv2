-- ===== GHOST HOP (Admin / Staff Detection) =====
local function ghostHop()
    print("⚠️ Admin detected! Ghost Hop Activated")
    local servers = HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Asc&limit=50"))
    for _, s in pairs(servers.data) do
        if s.playing < 5 and s.id ~= game.JobId then
            TeleportService:TeleportToPlaceInstance(game.PlaceId, s.id, player)
            break
        end
    end
end

    end
end)

-- ===== ATTACK LOGIC =====
local function doAttack(target)
    if target and target:FindFirstChild("HumanoidRootPart") then
        local randDelay = math.random(30,80)/1000
        task.wait(randDelay)
        ReplicatedStorage.Remotes.Validator:FireServer(tick(),math.random(100,999))
        ReplicatedStorage.Remotes.CommF_:InvokeServer("AttackRemote", target.HumanoidRootPart.CFrame)
    end
end

task.spawn(function()
    while task.wait(0.05) do
        if getFastAttack() or getKillAura() then
            for _, enemy in pairs(workspace.Enemies:GetChildren()) do
                if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
                    hrp.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0,0,5)
                    doAttack(enemy)
                end
            end
        end
    end
end)
