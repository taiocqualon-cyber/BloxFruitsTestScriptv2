-- ===== FUNCTION TẠO TOGGLE CHO TỪNG TỘC =====
local function CreateTribeToggle(name, parent, callback)
    local frame = Instance.new("Frame")
    frame.Parent = parent
    frame.Size = UDim2.new(1,0,0,35)
    frame.BackgroundTransparency = 1
    frame.Position = UDim2.new(0,0,0,#parent:GetChildren()/2*0.08)
    
    local label = Instance.new("TextLabel")
    label.Parent = frame
    label.Size = UDim2.new(0.7,0,1,0)
    label.BackgroundTransparency = 1
    label.Text = name
    label.TextColor3 = Color3.fromRGB(255,255,255)
    label.TextScaled = true
    label.Font = Enum.Font.SourceSans
    
    local toggle = Instance.new("TextButton")
    toggle.Parent = frame
    toggle.Size = UDim2.new(0.3,0,1,0)
    toggle.Position = UDim2.new(0.7,0,0,0)
    toggle.BackgroundColor3 = Color3.fromRGB(50,50,50)
    toggle.TextColor3 = Color3.fromRGB(255,255,255)
    toggle.Text = "OFF"
    
    local active = false
    toggle.MouseButton1Click:Connect(function()
        active = not active
        toggle.Text = active and "ON" or "OFF"
        if callback then
            pcall(callback,active)
        end
    end)
end
