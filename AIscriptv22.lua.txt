-- Hàm tạo Nút Chức Năng (Toggle)
local function addToggle(name)
    local frame = Instance.new("Frame", contentFrame)
    frame.Size = UDim2.new(1, -20, 0, 45)
    frame.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
    Instance.new("UICorner", frame)
    
    local btn = Instance.new("TextButton", frame)
    btn.Size = UDim2.new(1, 0, 1, 0)
    btn.BackgroundTransparency = 1
    btn.Text = "  " .. name .. " [OFF]"
    btn.Font = Enum.Font.Gotham
    btn.TextColor3 = Color3.new(0.8, 0.8, 0.8)
    btn.TextXAlignment = Enum.TextXAlignment.Left

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.Text = "  " .. name .. (active and " [ON]" or " [OFF]")
        btn.TextColor3 = active and Color3.fromRGB(0, 255, 200) or Color3.new(0.8, 0.8, 0.8)
    end)
    return function() return active end
end

local function doAttack(target)
    if target and target:FindFirstChild("Humanoid") then
        local remotes = ReplicatedStorage:WaitForChild("Remotes")
        -- Gửi Validator giả lập Ping thật
        remotes.Validator:FireServer(tick(), math.random(100, 999))
        remotes.CommF_:InvokeServer("AttackRemote", target.HumanoidRootPart.CFrame)
    end
end
