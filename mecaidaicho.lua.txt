local function Hop()
    local PlaceId = game.PlaceId
    local ApiUrl = "https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
    local success, result = pcall(function() return HttpService:JSONDecode(game:HttpGet(ApiUrl)) end)
    if success and result.data then
        for _, s in ipairs(result.data) do
            if s.playing < s.maxPlayers and s.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(PlaceId, s.id, Player)
                break
            end
        end
    end
end

local function CreateESP(part, name, color)
    if not part then return end
    local bgu = Instance.new("BillboardGui", CoreGui)
    bgu.Name = "ESP_UI"
    bgu.AlwaysOnTop = true
    bgu.Size = UDim2.new(0, 100, 0, 50)
    bgu.Adornee = part
    local txt = Instance.new("TextLabel", bgu)
    txt.BackgroundTransparency = 1; txt.Size = UDim2.new(1, 0, 1, 0)
    txt.Text = name; txt.TextColor3 = color; txt.Font = "SourceSansBold"; txt.TextSize = 14
    local highlight = Instance.new("Highlight", part)
    highlight.FillColor = color; highlight.OutlineColor = Color3.new(1, 1, 1)
end

local function RefreshESP()
    for _, v in pairs(CoreGui:GetChildren()) do if v.Name == "ESP_UI" then v:Destroy() end end
    for _, v in pairs(game.Workspace:GetDescendants()) do if v:IsA("Highlight") then v:Destroy() end end
    
    -- ESP Fruits
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and (v.Name:find("Fruit") or v.Name:find("Trái")) then
            CreateESP(v:FindFirstChild("Handle") or v, "🍎 " .. v.Name, Color3.fromRGB(255, 0, 0))
        end
    end
    -- ESP NPCs
    local npcTargets = {
        ["Master of Auras"] = {Color3.fromRGB(255, 255, 0), "✨ Haki"},
        ["Legendary Sword Dealer"] = {Color3.fromRGB(0, 255, 255), "⚔️ Sword"},
        ["Colors Specialist"] = {Color3.fromRGB(255, 0, 255), "🎨 Color"}
    }
    for _, v in pairs(game.Workspace:GetDescendants()) do
        if npcTargets[v.Name] then
            CreateESP(v:FindFirstChild("HumanoidRootPart") or v, npcTargets[v.Name][2], npcTargets[v.Name][1])
        end
    end
end

local function MakeBtn(parent, txt, col, func)
    local b = Instance.new("TextButton", parent)
    b.Size = UDim2.new(0, 220, 0, 45)
    b.BackgroundColor3 = col
    b.Text = txt; b.TextColor3 = Color3.new(1, 1, 1); b.Font = "SourceSansBold"; b.TextSize = 14
    Instance.new("UICorner", b).CornerRadius = UDim.new(0, 8)
    b.MouseButton1Click:Connect(func)
end
