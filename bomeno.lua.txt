-- Delta-friendly Blox Fruits Legendary Fruit Finder
-- Chạy trong Delta Executor

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local LocalPlayer = Players.LocalPlayer
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local HttpService = game:GetService("HttpService")

-- Danh sách trái truyền thuyết
local LegendaryFruits = {
    "Dragon Fruit",
    "Phoenix Fruit",
    "Leopard Fruit",
    "Dark Fruit",
    "Light Fruit"
}

-- Kiểm tra có phải trái truyền thuyết không
local function isLegendary(fruitName)
    for _, name in pairs(LegendaryFruits) do
        if fruitName == name then
            return true
        end
    end
    return false
end

-- Hop server Delta-friendly
local function hopServer()
    local placeId = game.PlaceId
    local servers = {}

    local success, response = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..placeId.."/servers/Public?sortOrder=Asc&limit=100")
    end)
    
    if success then
        local data = HttpService:JSONDecode(response)
        for _, server in pairs(data.data) do
            if server.playing < server.maxPlayers then
                table.insert(servers, server.id)
            end
        end
    end

    if #servers > 0 then
        local randomServer = servers[math.random(1, #servers)]
        TeleportService:TeleportToPlaceInstance(placeId, randomServer, LocalPlayer)
    else
        print("Không tìm thấy server trống, thử lại sau 5s")
        wait(5)
        hopServer()
    end
end

-- Tự nhặt trái trên map (Delta-friendly)
local function pickupFruits()
    for _, fruit in pairs(workspace:GetDescendants()) do
        if fruit:IsA("Tool") and isLegendary(fruit.Name) then
            fruit.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame
            wait(0.5)
            firetouchinterest(fruit.Handle, LocalPlayer.Character.HumanoidRootPart, 0)
            firetouchinterest(fruit.Handle, LocalPlayer.Character.HumanoidRootPart, 1)
            print("Đã nhặt trái truyền thuyết: "..fruit.Name)
        end
    end
end

-- Vòng lặp chính
while true do
    pickupFruits()

    local backpackFruit = LocalPlayer.Backpack:FindFirstChildWhichIsA("Tool")
    if backpackFruit and isLegendary(backpackFruit.Name) then
        print("Đã có trái truyền thuyết: "..backpackFruit.Name.." ✅")
        break
    else
        print("Chưa tìm thấy trái truyền thuyết, hop server tiếp...")
        hopServer()
        wait(2)
    end
end
