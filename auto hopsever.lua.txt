-- ===== MOON CHECK =====
local function GetMoon()
	local t = Lighting.ClockTime
	if t >= 0 and t <= 2 then return 4, "🌕 FULL MOON"
	elseif t <= 5 then return 3, "🌖 GẦN FULL"
	else return 1, "🌘 / 🌗 CÒN XA" end
end

local function TimeToFull()
	local t = Lighting.ClockTime
	local remain = (24 - t) % 24
	return math.floor(remain*60).."s"
end

task.spawn(function()
	while true do
		local lvl, txt = GetMoon()
		status.Text = "Moon: "..txt.."\nNext: "..TimeToFull()
		task.wait(1)
	end
end)

-- ===== CINEMATIC MINI INTRO =====
local function SpawnIntroEffect()
	local aura = Instance.new("ParticleEmitter", hrp)
	aura.Rate = 150
	aura.Speed = NumberRange.new(5,10)
	aura.Lifetime = NumberRange.new(1,2)
	aura.Size = NumberSequence.new(3)
	aura.Color = ColorSequence.new(Color3.fromRGB(255,0,0))

	local particles = {}
	for i=1,6 do
		local p = Instance.new("Part", workspace)
		p.Size = Vector3.new(1,1,1)
		p.Anchored = true
		p.Material = Enum.Material.Neon
		p.Color = Color3.fromRGB(100,0,0)
		p.CFrame = hrp.CFrame * CFrame.new(math.random(-3,3),0,math.random(-3,3))
		local pe = Instance.new("ParticleEmitter", p)
		pe.Rate = 150
		pe.Speed = NumberRange.new(2,5)
		pe.Lifetime = NumberRange.new(1,2)
		pe.Size = NumberSequence.new(2)
		pe.Color = ColorSequence.new(Color3.fromRGB(100,0,0))
		table.insert(particles, p)
	end
	task.delay(5, function()
		aura:Destroy()
		for _,p in pairs(particles) do
			if p then p:Destroy() end
		end
	end)
end

-- ===== HOP SERVER BUTTON =====
local visited = {}
btn.MouseButton1Click:Connect(function()
	status.Text = "🔎 GOD AI đang tìm server 🌖🌕..."
	SpawnIntroEffect()
	task.spawn(function()
		local success = false
		local req = game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=50")
		local data = HttpService:JSONDecode(req)
		for _,v in pairs(data.data or {}) do
			if v.playing < v.maxPlayers and not visited[v.id] then
				visited[v.id] = true
				success = true
				status.Text = "⏩ Hop tới server: "..v.id
				TeleportService:TeleportToPlaceInstance(game.PlaceId, v.id, player)
				break
			end
		end
		if not success then
			status.Text = "❌ Không tìm thấy server tốt"
		end
	end)
end)
