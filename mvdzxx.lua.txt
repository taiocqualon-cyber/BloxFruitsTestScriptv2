-- ===== ANIMATION =====
local open = false

	open = not open
	
	TweenService:Create(frame, TweenInfo.new(0.4), {
		Position = open and UDim2.new(0.02,0,0.1,0) or UDim2.new(0,-350,0.1,0)
	}):Play()
end)

-- ===== MOON CHECK =====
local function GetMoon()
	local t = Lighting.ClockTime
	if t >= 0 and t <= 2 then
		return 4, "🌕 FULL MOON"
	elseif t <= 5 then
		return 3, "🌖 GẦN FULL"
	elseif t <= 12 then
		return 2, "🌗 TRUNG"
	else
		return 1, "🌘 XA"
	end
end

local function TimeToFull()
	local t = Lighting.ClockTime
	local remain = (24 - t) % 24
	return math.floor(remain*60).."s"
end

-- ===== UPDATE UI =====
task.spawn(function()
	while true do
		local lvl, txt = GetMoon()
		status.Text = "Moon: "..txt..
		"\nNext: "..TimeToFull()
		task.wait(1)
	end
end)

-- ===== AI SERVER MEMORY =====
local visited = {}

-- ===== FIND SERVER =====
btn.MouseButton1Click:Connect(function()
	status.Text = "🔎 AI đang scan server..."

	local checked = 0

	for i = 1,30 do
		checked += 1

		local lvl, txt = GetMoon()

		if lvl == 4 then
			status.Text = "🌕 FULL MOON FOUND!"
			return
		elseif lvl == 3 then
			status.Text = "🌖 SERVER GOOD!"
			return
		end

		status.Text = "🔄 Server "..checked.." không tốt..."

		local req = game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
		local data = HttpService:JSONDecode(req)

		for _,v in pairs(data.data) do
			if v.playing < v.maxPlayers and not visited[v.id] then
				visited[v.id] = true
				TeleportService:TeleportToPlaceInstance(game.PlaceId, v.id, player)
				break
			end
		end

		task.wait(1.5)
	end

	status.Text = "❌ Không tìm thấy server tốt!"
end)
