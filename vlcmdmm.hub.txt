loadstring([[
repeat task.wait() until game:IsLoaded()

local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")

local player = Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local root = char:WaitForChild("HumanoidRootPart")

-- GUI
local gui = Instance.new("ScreenGui", player.PlayerGui)

-- NÚT TRÒN
local open = Instance.new("TextButton", gui)
open.Size = UDim2.new(0,60,0,60)
open.Position = UDim2.new(0.05,0,0.4,0)
open.Text = "HUB"
open.BackgroundColor3 = Color3.fromRGB(0,170,255)
open.TextColor3 = Color3.new(1,1,1)
open.Draggable = true
Instance.new("UICorner", open).CornerRadius = UDim.new(1,0)

-- MAIN
local main = Instance.new("Frame", gui)
main.Size = UDim2.new(0,300,0,200)
main.Position = UDim2.new(0.3,0,0.3,0)
main.BackgroundColor3 = Color3.fromRGB(25,25,25)
main.Visible = false
main.Active = true
main.Draggable = true
Instance.new("UICorner", main)

-- TITLE
local title = Instance.new("TextLabel", main)
title.Size = UDim2.new(1,0,0,30)
title.Text = "Chest Hub Pro 😎"
title.TextColor3 = Color3.new(1,1,1)
title.BackgroundTransparency = 1

-- TOGGLE
local toggle = Instance.new("TextButton", main)
toggle.Size = UDim2.new(0,200,0,40)
toggle.Position = UDim2.new(0,50,0,80)
toggle.Text = "Auto Chest: OFF"
toggle.BackgroundColor3 = Color3.fromRGB(70,70,70)

local running = false

toggle.MouseButton1Click:Connect(function()
	running = not running
	toggle.Text = "Auto Chest: "..(running and "ON" or "OFF")
end)

-- MỞ MENU
open.MouseButton1Click:Connect(function()
	main.Visible = not main.Visible
end)

-- DANH SÁCH ĐẢO
local islands = {
	CFrame.new(-1600, 10, 150), -- Jungle
	CFrame.new(1100, 10, -1400), -- Snow
	CFrame.new(-5200, 10, 8500), -- Magma
	CFrame.new(-4800, 700, -2600), -- Sky 1
	CFrame.new(-7900, 5500, -380), -- Sky 2
	CFrame.new(-7900, 5600, -1400), -- Sky 3
}

local islandIndex = 1
local collected = {}

-- TWEEN
function tweenTo(cf)
	local dist = (root.Position - cf.Position).Magnitude
	local speed = 220

	local tween = TweenService:Create(
		root,
		TweenInfo.new(dist/speed, Enum.EasingStyle.Linear),
		{CFrame = cf}
	)

	tween:Play()
	tween.Completed:Wait()
end

-- TÌM RƯƠNG
function getChest()
	local dist = math.huge
	local chosen = nil

	for _,v in pairs(workspace:GetDescendants()) do
		if v:IsA("BasePart") and string.find(v.Name:lower(),"chest") then
			if not collected[v] then
				local d = (root.Position - v.Position).Magnitude
				if d < dist then
					dist = d
					chosen = v
				end
			end
		end
	end

	return chosen
end

-- LOOP
spawn(function()
	while true do
		task.wait(0.3)

		if running then
			local chest = getChest()

			if chest then
				tweenTo(chest.CFrame + Vector3.new(0,3,0))
				task.wait(1)
				collected[chest] = true
			else
				-- reset và qua đảo
				collected = {}
				tweenTo(islands[islandIndex])
				task.wait(2)

				islandIndex = islandIndex + 1
				if islandIndex > #islands then
					islandIndex = 1
				end
			end
		end
	end
end)

]])()
