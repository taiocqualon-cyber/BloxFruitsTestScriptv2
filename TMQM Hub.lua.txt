-- ===== FRUIT LIST =====
local GOD = {
	Leopard=true,Dragon=true,Kitsune=true,
	Dough=true,Venom=true,Spirit=true
}

local function isGod(name)
	for k,_ in pairs(GOD) do
		if string.find(string.lower(name),string.lower(k)) then
			return true
		end
	end
	return false
end

-- ===== ESP ALL FRUITS =====
local function ESP(obj, isGodFruit)
	if obj:FindFirstChild("ESP") then return end

	local bill = Instance.new("BillboardGui", obj)
	bill.Name = "ESP"
	bill.Size = UDim2.new(0,120,0,40)
	bill.AlwaysOnTop = true

	local txt = Instance.new("TextLabel", bill)
	txt.Size = UDim2.new(1,0,1,0)
	txt.BackgroundTransparency = 1
	txt.TextScaled = true

	if isGodFruit then
		txt.TextColor3 = Color3.fromRGB(255,0,0)
		txt.Text = "🔥 "..obj.Name
	else
		txt.TextColor3 = Color3.fromRGB(0,255,0)
		txt.Text = obj.Name
	end
end

-- ===== TP =====
local function tp(pos)
	local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
	if root then root.CFrame = CFrame.new(pos) end
end

-- ===== MAIN =====
task.spawn(function()
	while true do
		task.wait(2)

		for _,v in pairs(workspace:GetDescendants()) do
			if v:IsA("Tool") and v:FindFirstChild("Handle") then
				
				local god = isGod(v.Name)
				ESP(v, god)

				if autoFruit and god then
					content.Text = "🔥 "..v.Name
					tp(v.Handle.Position + Vector3.new(0,2,0))
				end
			end
		end
	end
end)

	)

		if v.playing >= 8 and v.playing < v.maxPlayers then
			TeleportService:TeleportToPlaceInstance(game.PlaceId,v.id,player)
			break
		end
	end
