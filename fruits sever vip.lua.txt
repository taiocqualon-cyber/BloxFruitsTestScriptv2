repeat task.wait() until game:IsLoaded()

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")

local player = Players.LocalPlayer
local PlayerGui = player:WaitForChild("PlayerGui")
local placeId = game.PlaceId

-- ================= GIÁ TRÁI =================
local fruitPrices = {
	["dragon"]=3500000,
	["tiger"]=5000000,
	["kitsune"]=8000000,
	["dough"]=2800000,
	["venom"]=3000000,
	["spirit"]=3400000,
	["control"]=3200000,
	["shadow"]=2900000,
	["t-rex"]=2700000,
	["mammoth"]=2600000,
	["gravity"]=2500000,
	["blizzard"]=2400000,
	["portal"]=1900000,
	["pain"]=2300000,
	["gas"]=2100000,
}

-- ================= GUI =================
local gui = Instance.new("ScreenGui", PlayerGui)

-- mũi tên
local arrow = Instance.new("TextLabel", gui)
arrow.Size = UDim2.new(0,320,0,50)
arrow.Position = UDim2.new(0.5,-160,0.85,0)
arrow.BackgroundTransparency = 1
arrow.TextColor3 = Color3.fromRGB(0,255,0)
arrow.TextScaled = true
arrow.Text = "Scanning..."

-- nút hop
local hopBtn = Instance.new("TextButton", gui)
hopBtn.Size = UDim2.new(0,200,0,50)
hopBtn.Position = UDim2.new(0.5,-100,0.75,0)
hopBtn.Text = "🔁 ĐỔI SERVER ÍT NGƯỜI"
hopBtn.BackgroundColor3 = Color3.fromRGB(255,50,50)
hopBtn.TextColor3 = Color3.new(1,1,1)
hopBtn.Visible = false

-- âm thanh
local sound = Instance.new("Sound", gui)
sound.SoundId = "rbxassetid://9118828563"
sound.Volume = 1.5

-- ================= CHECK =================
local function getFruitValue(v)
	local name = v.Name:lower()
	for fruit,price in pairs(fruitPrices) do
		if name == fruit or name:match("^"..fruit) then
			return price
		end
	end
end

local function getPos(v)
	if v:IsA("Tool") and v:FindFirstChild("Handle") then
		return v.Handle.Position
	elseif v:IsA("BasePart") then
		return v.Position
	end
end

-- ================= SERVER ÍT NGƯỜI =================
local function getServer()
	local url = "https://games.roblox.com/v1/games/"..placeId.."/servers/Public?sortOrder=Asc&limit=100"
	local data = HttpService:JSONDecode(game:HttpGet(url))
	
	for _,v in pairs(data.data) do
		if v.playing < v.maxPlayers then
			return v.id
		end
	end
end

-- ================= BIẾN =================
local bestFruit = nil
local bestPrice = 0

-- ================= QUÉT =================
task.spawn(function()
	while true do
		task.wait(2)
		
		bestFruit = nil
		bestPrice = 0
		
		for _,v in pairs(workspace:GetDescendants()) do
			local price = getFruitValue(v)
			
			if price and price > bestPrice then
				bestPrice = price
				bestFruit = v
			end
		end
		
		if bestFruit then
			hopBtn.Visible = false
			sound:Play()
		else
			hopBtn.Visible = true
		end
	end
end)

-- ================= HIỂN THỊ =================
task.spawn(function()
	while true do
		task.wait(0.3)
		
		local char = player.Character
		if not char or not char:FindFirstChild("HumanoidRootPart") then continue end
		
		local root = char.HumanoidRootPart.Position
		
		if bestFruit then
			local pos = getPos(bestFruit)
			
			if pos then
				local dist = (pos - root).Magnitude
				
				arrow.Text = "🎯 "..bestFruit.Name..
				" ("..math.floor(bestPrice/1000000).."M) - "..math.floor(dist).."m"
			end
		else
			arrow.Text = "❌ Không có trái → đổi server"
		end
	end
end)

-- ================= CLICK HOP =================
hopBtn.MouseButton1Click:Connect(function()
	local id = getServer()
	
	if id then
		TeleportService:TeleportToPlaceInstance(placeId, id, player)
	else
		TeleportService:Teleport(placeId)
	end
end)
