-- 👹 AUTO HOP FRUIT XỊN (STABLE)

repeat wait() until game:IsLoaded()

local Players = game:GetService("Players")
local plr = Players.LocalPlayer
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")

-- 🎯 LIST TRÁI XỊN
local goodFruits = {
    ["Kitsune"] = true,
    ["Dough"] = true,
    ["Dragon"] = true,
    ["Leopard"] = true,
    ["Spirit"] = true,
    ["Venom"] = true,
    ["Control"] = true
}

-- ===== CHECK FRUIT =====
local function isFruit(v)
    return v:IsA("Tool") and v:FindFirstChild("Handle") and v.Parent == workspace
end

local function isGoodFruit(v)
    for name,_ in pairs(goodFruits) do
        if string.find(v.Name, name) then
            return true
        end
    end
    return false
end

-- ===== SERVER HOP =====
local function changeServer()
    local servers = {}

    local ok, res = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
    end)

    if ok then
        local data = HttpService:JSONDecode(res)
        for _,s in pairs(data.data) do
            if s.playing < s.maxPlayers-2 then
                table.insert(servers, s.id)
            end
        end
    end

    if #servers > 0 then
        TeleportService:TeleportToPlaceInstance(
            game.PlaceId,
            servers[math.random(1,#servers)],
            plr
        )
    end
end

-- ===== LOOP =====
while true do
    local foundGood = false
    local foundAny = false

    for _,v in pairs(workspace:GetDescendants()) do
        if isFruit(v) then
            foundAny = true

            if isGoodFruit(v) then
                foundGood = true

                warn("🔥 FOUND GOOD FRUIT: "..v.Name)

                -- TP NHẶT
                local hrp = plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    hrp.CFrame = v.Handle.CFrame + Vector3.new(0,3,0)
                end

                break
            end
        end
    end

    -- ❌ KHÔNG CÓ TRÁI
    if not foundAny then
        warn("❌ No fruit → đổi server")
        changeServer()
        wait(3)

    -- 🍈 CÓ TRÁI NHƯNG KHÔNG XỊN
    elseif not foundGood then
        warn("🍈 Fruit thường → đổi server")
        changeServer()
        wait(3)

    else
        warn("💀 Giữ server (có trái xịn)")
        break
    end

    wait(1)
end
