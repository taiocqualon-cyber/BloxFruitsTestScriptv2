repeat wait() until game:IsLoaded()

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")

local plr = Players.LocalPlayer

-- ===== GUI =====
local gui = Instance.new("ScreenGui")
gui.Parent = plr:WaitForChild("PlayerGui")

local status = Instance.new("TextLabel", gui)
status.Size = UDim2.new(0,250,0,40)
status.Position = UDim2.new(0,10,0.5,0)
status.BackgroundColor3 = Color3.fromRGB(0,0,0)
status.TextColor3 = Color3.fromRGB(0,255,0)
status.TextScaled = true
status.Text = "🔄 Đang khởi động..."

-- ===== TRÁI MYTHIC =====
local mythicList = {
    ["Kitsune"]=true,
    ["Dragon"]=true,
    ["Leopard"]=true,
    ["Dough"]=true,
    ["Spirit"]=true,
    ["Venom"]=true,
    ["Control"]=true
}

-- ===== HÀM =====
local function setStatus(txt)
    status.Text = txt
    warn(txt)
end

local function isFruit(v)
    return v:IsA("Tool") and v:FindFirstChild("Handle") and v.Parent == workspace
end

local function isMythic(v)
    for name,_ in pairs(mythicList) do
        if string.find(v.Name,name) then
            return true
        end
    end
    return false
end

-- ===== HOP SERVER =====
local function hopServer()
    setStatus("🔁 Đang đổi server...")

    local ok,res = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
    end)

    if ok then
        local data = HttpService:JSONDecode(res)
        local servers = {}

        for _,s in pairs(data.data) do
            if s.playing < s.maxPlayers - 1 then
                table.insert(servers, s.id)
            end
        end

        if #servers > 0 then
            local id = servers[math.random(1,#servers)]

            wait(1)
            TeleportService:TeleportToPlaceInstance(
                game.PlaceId,
                id,
                plr
            )
        else
            setStatus("⏳ Không có server → đợi")
            wait(5)
        end
    else
        setStatus("❌ Lỗi lấy server → thử lại")
        wait(5)
    end
end

-- ===== MAIN LOOP =====
while true do
    wait(3)

    setStatus("🔍 Đang tìm trái...")

    local foundMythic = false

    for _,v in pairs(workspace:GetDescendants()) do
        if isFruit(v) and isMythic(v) then
            foundMythic = true

            setStatus("🔥 TÌM THẤY: "..v.Name)

            local char = plr.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                char.HumanoidRootPart.CFrame =
                    v.Handle.CFrame + Vector3.new(0,3,0)
            end

            break
        end
    end

    if not foundMythic then
        hopServer()
    else
        setStatus("👹 Đã nhặt trái → dừng")
        break
    end
end
