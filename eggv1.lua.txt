-- [[ HÀM NHẢY SERVER VẮNG (1-3 NGƯỜI) ]] --
function HopSmallServer()
    Rayfield:Notify({Title = "Hệ thống", Content = "Đang tìm server vắng người (1-3 người)...", Duration = 3})
    local Http = game:GetService("HttpService")
    local Api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
    
    local success, result = pcall(function()
        local Raw = game:HttpGet(Api)
        local Servers = Http:JSONDecode(Raw).data
        for _, s in pairs(Servers) do
            if s.playing >= 1 and s.playing <= 3 and s.id ~= game.JobId then
                game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, s.id, game.Players.LocalPlayer)
                return
            end
        end
    end)
    
    if not success then 
        Rayfield:Notify({Title = "Lỗi", Content = "Không tìm thấy server vắng, đang thử lại...", Duration = 2})
        task.wait(1)
        HopSmallServer()
    end
end

-- [[ TAB 1: 🥚 ULTRA EGG HUNTER ]] --
local HunterTab = Window:CreateTab("🥚 Egg Hunter", 4483362458)

HunterTab:CreateToggle({
   Name = "Hiện Trứng & Trứng VIP (ESP)",
   CurrentValue = true,
   Callback = function(Value) _G.EggESP = Value end,
})

HunterTab:CreateButton({
   Name = "🚀 Bay tới Trứng gần nhất (Tween)",
   Callback = function()
      local target = nil
      local dist = math.huge
      for _, v in pairs(game.Workspace:GetChildren()) do
          if (v.Name:find("Egg") or v.Name:find("Easter")) and v:IsA("BasePart") then
              local d = (v.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
              if d < dist then dist = d; target = v end
          end
      end
      if target then
          game:GetService("TweenService"):Create(game.Players.LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(dist/_G.TweenSpeed), {CFrame = target.CFrame}):Play()
      else
          Rayfield:Notify({Title = "Thông báo", Content = "Server này đã hết Trứng!", Duration = 2})
      end
   end,
})

HunterTab:CreateButton({
   Name = "🌍 Nhảy Server Vắng (1-3 người)",
   Callback = function() HopSmallServer() end,
})

                    end

            end

end)
