local function createESP(fruit)
	if fruit:FindFirstChild("Handle") then
		
		local esp = Instance.new("BillboardGui")
		esp.Name = "FruitESP"
		esp.Size = UDim2.new(0,120,0,40)
		esp.Adornee = fruit.Handle
		esp.AlwaysOnTop = true
		esp.StudsOffset = Vector3.new(0,2,0)

		-- 🔵 CHẤM TRÒN
		local dot = Instance.new("Frame", esp)
		dot.Size = UDim2.new(0,12,0,12)
		dot.Position = UDim2.new(0.5,-6,0,0)
		dot.BorderSizePixel = 0
		
		local corner = Instance.new("UICorner", dot)
		corner.CornerRadius = UDim.new(1,0)

		-- 📝 TEXT
		local text = Instance.new("TextLabel", esp)
		text.Size = UDim2.new(1,0,0,20)
		text.Position = UDim2.new(0,0,0,14)
		text.BackgroundTransparency = 1
		text.TextScaled = true
		text.TextStrokeTransparency = 0
		text.TextColor3 = Color3.new(1,1,1)
		text.Text = "🍏 Fruits Blox"

		-- ===== MÀU THEO ĐỘ HIẾM =====
		local price = fruitPrices[fruit.Name:lower()] or 0
		
		if price < 2000000 then
			dot.BackgroundColor3 = Color3.fromRGB(255,255,255) -- trắng
		elseif price < 3000000 then
			dot.BackgroundColor3 = Color3.fromRGB(170,0,255) -- tím
		elseif price < 5000000 then
			dot.BackgroundColor3 = Color3.fromRGB(255,255,0) -- vàng
		else
			dot.BackgroundColor3 = Color3.fromRGB(255,0,0) -- đỏ
		end

		esp.Parent = fruit

		-- 🔁 UPDATE KHOẢNG CÁCH
		task.spawn(function()
			while fruit.Parent and esp.Parent do
				task.wait(0.3)

				local char = player.Character
				if char and char:FindFirstChild("HumanoidRootPart") then
					local dist = (fruit.Handle.Position - char.HumanoidRootPart.Position).Magnitude
					
					text.Text = "🍏 Fruits Blox - "..math.floor(dist).."m"
				end
			end
		end)
	end
end
