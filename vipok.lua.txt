-- ===== MOON CHECK =====
local function GetMoon()
	local t = Lighting.ClockTime
	if t >= 0 and t <= 2 then
		return 4, "🌕 FULL MOON"
	elseif t <= 5 then
		return 3, "🌖 GẦN FULL"
	else
		return 1, "🌘 / 🌗 CÒN XA"
	end
end

local function TimeToFull()
	local t = Lighting.ClockTime
	local remain = (24 - t) % 24
	return math.floor(remain*60).."s"
end

-- Update UI realtime
task.spawn(function()
	while true do
		local lvl, txt = GetMoon()
		status.Text = "Moon: "..txt.."\nNext: "..TimeToFull()
		task.wait(1)
	end
end)

-- ===== AI HIGH CHANCE 🌖🌕 =====
local visited = {}
btn.MouseButton1Click:Connect(function()
	status.Text = "🔎 GOD AI đang scan server..."
	task.spawn(function()
		for attempt=1,50 do
			if attempt>1 then task.wait(0.3) end
			local lvl, txt = GetMoon()
			if lvl >= 3 then
				status.Text = "✅ Server tốt: "..txt
				return
			end

			status.Text = "🔄 Hop server "..attempt.."..."
			local req = game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=50")
			local data = HttpService:JSONDecode(req)

			for _,v in pairs(data.data) do
				if v.playing < v.maxPlayers and not visited[v.id] then
					visited[v.id] = true
					TeleportService:TeleportToPlaceInstance(game.PlaceId, v.id, player)
					break
				end
			end
		end
		status.Text = "❌ Không tìm thấy server tốt!"
	end)
end)
