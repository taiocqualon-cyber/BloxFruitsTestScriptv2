local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")

local player = Players.LocalPlayer
local gui = Instance.new("ScreenGui", player.PlayerGui)

-- MENU CHÍNH
local menu = Instance.new("Frame", gui)
menu.Size = UDim2.new(0,150,0,60)
menu.Position = UDim2.new(0.5,-75,0.5,-30)
menu.AnchorPoint = Vector2.new(0.5,0.5)
menu.BackgroundColor3 = Color3.fromRGB(0,50,50)
menu.Visible = false
menu.ClipsDescendants = true
local corner = Instance.new("UICorner", menu)
corner.CornerRadius = UDim.new(0.3,0)

-- NÚT ON/OFF
local toggle = Instance.new("TextButton", menu)
toggle.Size = UDim2.new(0,120,0,40)
toggle.Position = UDim2.new(0.5,-60,0.5,-20)
toggle.Text = "OFF"
toggle.BackgroundColor3 = Color3.fromRGB(0,170,255)
toggle.TextColor3 = Color3.new(1,1,1)
toggle.Font = Enum.Font.GothamBold

local running = false
toggle.MouseButton1Click:Connect(function()
	running = not running
	toggle.Text = running and "ON" or "OFF"

	if running then
		-- FIX LAG EFFECT
		local Lighting = game:GetService("Lighting")
		Lighting.GlobalShadows = false
		Lighting.FogEnd = 100000
		Lighting.Brightness = 1

		-- Giảm 30% hiệu ứng
		for _,v in pairs(Lighting:GetDescendants()) do
			if v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") then
				v.Enabled = false
			end
		end

		-- Giảm vật liệu / decal
		for _,v in pairs(workspace:GetDescendants()) do
			if v:IsA("BasePart") then
				v.Material = Enum.Material.SmoothPlastic
				v.Reflectance = 0
			elseif v:IsA("Texture") or v:IsA("Decal") then
				v.Transparency = 1
			end
		end

		-- Thông báo
		print("✅ FIX LAG VIP ACTIVATED")
	end
end)

-- INTRO HIỆU ỨNG CHỮ
local text = "Fix Lag"
local letters = {}
local startX = 0.2

for i = 1, #text do
	local char = text:sub(i,i)
	local lbl = Instance.new("TextLabel", gui)
	lbl.Text = char
	lbl.Font = Enum.Font.GothamBold
	lbl.TextColor3 = Color3.new(1,1,1)
	lbl.TextScaled = true
	lbl.Size = UDim2.new(0,50,0,50)
	lbl.Position = UDim2.new(startX + i*0.05,0,0.4,0)
	lbl.BackgroundTransparency = 1
	table.insert(letters,lbl)

	local tween = TweenService:Create(
		lbl,
		TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.Out),
		{TextTransparency = 0}
	)
	tween:Play()
	task.wait(0.15)
end

task.wait(1.5)

-- Hiện menu tròn
menu.Visible = true
