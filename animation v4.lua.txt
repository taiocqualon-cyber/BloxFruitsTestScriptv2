local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local hrp = char:WaitForChild("HumanoidRootPart")
local cam = workspace.CurrentCamera
local TweenService = game:GetService("TweenService")
local Lighting = game:GetService("Lighting")
local RunService = game:GetService("RunService")

-- ===== RESET CAMERA =====
cam.CameraType = Enum.CameraType.Scriptable

-- ===== 0️⃣ Đặt nhân vật + hover ổn định =====
local hoverTween = TweenService:Create(hrp, TweenInfo.new(5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
	CFrame = hrp.CFrame + Vector3.new(0,1,0)
})
hoverTween:Play()

-- ===== 1️⃣ Camera đối diện nhân vật =====
cam.CFrame = CFrame.new(hrp.Position + Vector3.new(0,5,10), hrp.Position)

-- ===== 2️⃣ Particle aura bay vòng quanh =====
local aura = Instance.new("ParticleEmitter", hrp)
aura.Rate = 80
aura.Lifetime = NumberRange.new(1,2)
aura.Speed = NumberRange.new(1,3)
aura.Size = NumberSequence.new(2)
aura.Color = ColorSequence.new({
	ColorSequenceKeypoint.new(0, Color3.fromRGB(255,0,0)),
	ColorSequenceKeypoint.new(1, Color3.fromRGB(255,215,0))
})

-- ===== 3️⃣ Flash đỏ server 3 giây =====
task.spawn(function()
	for i = 1,60 do
		local t = i/60
		Lighting.ColorShift_Top = Color3.fromRGB(255, t*255, t*0)
		Lighting.ColorShift_Bottom = Color3.fromRGB(255, t*215, 0)
		task.wait(0.05)
	end
end)

task.wait(3)

-- ===== 4️⃣ HIỆN CHỮ TRIỂN KHAI LÃNH ĐỊA =====
local gui = Instance.new("BillboardGui", hrp)
gui.Size = UDim2.new(0,300,0,100)
gui.Adornee = hrp
gui.AlwaysOnTop = true

local label = Instance.new("TextLabel", gui)
label.Size = UDim2.new(1,0,1,0)
label.BackgroundTransparency = 1
label.TextScaled = true
label.TextColor3 = Color3.fromRGB(255,0,0)
label.Text = "Triển khai lãnh địa 😈"

task.wait(2)
gui:Destroy()

-- ===== 5️⃣ Camera zoom chậm từ 5m → cận mặt 1m =====
local startCam = hrp.Position + Vector3.new(0,5,5)
local endCam = hrp.Position + Vector3.new(0,2,1)
TweenService:Create(cam, TweenInfo.new(3), {CFrame = CFrame.new(endCam, hrp.Position)}):Play()
task.wait(3)

-- ===== 6️⃣ Bay vòng quanh + hover =====
task.spawn(function()
	for i = 1,180 do
		RunService.RenderStepped:Wait()
		local rad = math.rad(i*2)
		local offset = Vector3.new(math.cos(rad)*3,1,math.sin(rad)*3)
		hrp.CFrame = CFrame.new(hrp.Position + offset)
	end
end)

-- ===== 7️⃣ TELEPORT SANG DOMAIN RIÊNG =====
task.wait(2)
local domainCenter = Vector3.new(1000,100,1000)
hrp.CFrame = CFrame.new(domainCenter + Vector3.new(0,2,0))

-- ===== 8️⃣ TỐI TOÀN BỘ DOMAIN + LÃNH ĐỊA 30M =====
Lighting.Ambient = Color3.fromRGB(0,0,0)
Lighting.OutdoorAmbient = Color3.fromRGB(0,0,0)

local domainFloor = Instance.new("Part", workspace)
domainFloor.Size = Vector3.new(60,1,60)
domainFloor.Anchored = true
domainFloor.Material = Enum.Material.Neon
domainFloor.Color = Color3.fromRGB(0,0,0)
domainFloor.Position = domainCenter - Vector3.new(0,0.5,0)

-- ===== 9️⃣ Mắt đỏ + hover liên tục =====
local eyes = Instance.new("PointLight", hrp)
eyes.Color = Color3.fromRGB(255,0,0)
eyes.Range = 5
eyes.Brightness = 2

TweenService:Create(hrp, TweenInfo.new(3, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
	CFrame = hrp.CFrame + Vector3.new(0,2,0)
}):Play()

-- ===== 10️⃣ Camera xoay quanh nhân vật liên tục =====
task.spawn(function()
	while true do
		RunService.RenderStepped:Wait()
		local t = tick()
		local offset = Vector3.new(math.cos(t)*15,5,math.sin(t)*15)
		cam.CFrame = CFrame.new(hrp.Position + offset, hrp.Position)
	end
end)

-- ===== 11️⃣ Particle nổ nhẹ xung quanh =====
task.spawn(function()
	while true do
		task.wait(0.2)
		local part = Instance.new("Part", workspace)
		part.Size = Vector3.new(1,1,1)
		part.Position = hrp.Position
		part.Anchored = false
		part.Material = Enum.Material.Neon
		part.Color = Color3.fromRGB(255,0,0)
		part.Velocity = Vector3.new(math.random(-20,20), math.random(20,40), math.random(-20,20))
		task.delay(2, function() part:Destroy() end)
	end
end)
