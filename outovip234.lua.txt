repeat task.wait() until game:IsLoaded()

local player = game.Players.LocalPlayer
local root = player.Character:WaitForChild("HumanoidRootPart")

-- GUI
local gui = Instance.new("ScreenGui", player.PlayerGui)

local btn = Instance.new("TextButton", gui)
btn.Size = UDim2.new(0,180,0,40)
btn.Position = UDim2.new(0.05,0,0.5,0)
btn.Text = "Next Island"
btn.BackgroundColor3 = Color3.fromRGB(0,170,255)

-- DANH SÁCH ĐẢO (có thể chỉnh lại cho đúng map bạn)
local islands = {
	{name="Jungle", pos=CFrame.new(-1600, 10, 150)}, 
	{name="Snow", pos=CFrame.new(1100, 10, -1400)},
	{name="Magma", pos=CFrame.new(-5200, 10, 8500)},
	{name="Sky 1", pos=CFrame.new(-4800, 700, -2600)},
	{name="Sky 2", pos=CFrame.new(-7900, 5500, -380)},
	{name="Sky 3", pos=CFrame.new(-7900, 5600, -1400)},
}

local index = 1

btn.MouseButton1Click:Connect(function()
	local island = islands[index]
	root.CFrame = island.pos
	btn.Text = "Island: "..island.name

	index = index + 1
	if index > #islands then
		index = 1
	end
end)
