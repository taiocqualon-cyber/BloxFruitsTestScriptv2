-- Toggle function
local function AddToggle(name, callback, ypos)
    local f = Instance.new("Frame", frame)
    f.Size = UDim2.new(1,-10,0,35)
    f.Position = UDim2.new(0,5,0,40 + (ypos-1)*40)
    f.BackgroundTransparency = 1

    local label = Instance.new("TextLabel", f)
    label.Size = UDim2.new(0.7,0,1,0)
    label.Text = name
    label.TextColor3 = Color3.fromRGB(255,255,255)
    label.BackgroundTransparency = 1
    label.TextScaled = true
    label.Font = Enum.Font.SourceSans

    local toggle = Instance.new("TextButton", f)
    toggle.Size = UDim2.new(0.3,0,1,0)
    toggle.Position = UDim2.new(0.7,0,0,0)
    toggle.Text = "OFF"
    toggle.TextColor3 = Color3.fromRGB(255,255,255)
    toggle.BackgroundColor3 = Color3.fromRGB(50,50,50)

    local active = false
    toggle.MouseButton1Click:Connect(function()
        active = not active
        toggle.Text = active and "ON" or "OFF"
        if callback then pcall(callback, active) end
    end)
end

-- ===== Add tộc với hiệu ứng thức tỉnh =====
AddToggle("Tộc Quỷ V4", function(active)
    if active then
        -- 6 particle đen đỏ xoay quanh
        for i=1,6 do
            local p = Instance.new("Part", workspace)
            p.Size = Vector3.new(2,2,2)
            p.Anchored = true
            p.Material = Enum.Material.Neon
            p.Color = Color3.fromRGB(50,0,0)
            p.CFrame = hrp.CFrame * CFrame.new(math.random(-5,5),0,math.random(-5,5))
            local pe = Instance.new("ParticleEmitter", p)
            pe.Rate = 150
            pe.Speed = NumberRange.new(2,5)
            pe.Size = NumberSequence.new(2)
            pe.Color = ColorSequence.new(Color3.fromRGB(50,0,0))
            task.delay(5,function() p:Destroy() end)
        end
    end
end, 1)
