-- ===== HÀM HỖ TRỢ THÔNG MINH (INTELLIGENCE) =====

-- 1. Smart Scanner: Tìm quái tối ưu nhất
local function getSmartTarget(monsterName, spawnPos)
    local bestTarget = nil
    local minDistance = 300
    
    for _, e in pairs(workspace.Enemies:GetChildren()) do
        if e.Name == monsterName and e:FindFirstChild("Humanoid") and e.Humanoid.Health > 0 then
            local dist = (e.HumanoidRootPart.Position - spawnPos).Magnitude
            if dist < minDistance then
                -- Ưu tiên quái máu thấp để dứt điểm nhanh
                if not bestTarget or e.Humanoid.Health < bestTarget.Humanoid.Health then
                    bestTarget = e
                end
            end
        end
    end
    return bestTarget
end

-- 2. Auto Stats: Tự động nâng điểm (Melee/Defense tỉ lệ 3:2)
local function autoStats()
    local points = player.Data.StatsPoints.Value
    if points > 0 then
        local remotes = ReplicatedStorage:FindFirstChild("Remotes") and ReplicatedStorage.Remotes:FindFirstChild("CommF_")
        if remotes then
            remotes:InvokeServer("AddPoint", "Melee", math.floor(points * 0.6))
            remotes:InvokeServer("AddPoint", "Defense", math.floor(points * 0.4))
        end
    end
end

-- 3. Auto Equip: Tự động cầm vũ khí đấm (Melee)
local function equipWeapon()
    if not character:FindFirstChildOfClass("Tool") then
        for _, tool in pairs(player.Backpack:GetChildren()) do
            if tool:IsA("Tool") and (tool.ToolTip == "Melee" or tool.Name:find("Combat")) then
                character.Humanoid:EquipTool(tool)
                break
            end
        end
    end
end

-- 4. Anti-Stuck & Tween
local lastActionTime = tick()
local function safeTween(targetPos)
    local dist = (hrp.Position - targetPos).Magnitude
    if dist > 10 then
        TweenService:Create(hrp, TweenInfo.new(dist/150, Enum.EasingStyle.Linear), {CFrame = CFrame.new(targetPos)}):Play()
        lastActionTime = tick()
    end
end

-- ===== CHU TRÌNH ĐIỀU KHIỂN FINAL =====
task.spawn(function()
    while task.wait(0.15) do
        if running then
            -- Tự động hóa phụ trợ
            equipWeapon()
            autoStats()
            
            local myLv = player.Data.Level.Value
            local cq = nil
            for i = #Auto_Database, 1, -1 do
                if myLv >= Auto_Database[i].Level then cq = Auto_Database[i]; break end
            end
            
            if cq then
                local pgui = player:FindFirstChild("PlayerGui")
                local questActive = pgui and pgui:FindFirstChild("Main") and pgui.Main:FindFirstChild("Quest") and pgui.Main.Quest.Visible
                
                if not questActive then
                    -- Bay nhận nhiệm vụ (Timeout Anti-Stuck)
                    safeTween(cq.Pos + Vector3.new(0, 5, 0))
                    task.wait(0.8)
                    local commF = ReplicatedStorage:FindFirstChild("Remotes") and ReplicatedStorage.Remotes:FindFirstChild("CommF_")
                    if commF then commF:InvokeServer("StartQuest", cq.QuestName, 1) end
                else
                    -- Tìm mục tiêu thông minh
                    local target = getSmartTarget(cq.Monster, cq.Pos)
                    if target then
                        -- Bay trên đầu né chiêu + Gom quái + Đấm tầm xa
                        hrp.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 25, 0)
                        target.HumanoidRootPart.CFrame = hrp.CFrame * CFrame.new(0, -25, 0)
                        target.HumanoidRootPart.CanCollide = false
                        
                        -- Tấn công Overlord
                        local rem = ReplicatedStorage:FindFirstChild("Remotes")
                        if rem and rem:FindFirstChild("CommF_") then
                            rem.Validator:FireServer(tick(), 0)
                            rem.CommF_:InvokeServer("AttackRemote", target.HumanoidRootPart.CFrame)
                        end
                        if ReplicatedStorage:FindFirstChild("Modules") and ReplicatedStorage.Modules:FindFirstChild("Net") then
                            ReplicatedStorage.Modules.Net.Attack:FireServer()
                        end
                    else
                        -- Bay chờ tại bãi spawn
                        safeTween(cq.Pos + Vector3.new(0, 55, 0))
                    end
                end
            end
            
            -- Anti-Stuck: Nếu đứng yên quá 30s mà không có quái, Reset vị trí
            if tick() - lastActionTime > 30 then
                hrp.CFrame = hrp.CFrame * CFrame.new(0, 100, 0)
                lastActionTime = tick()
            end
        end
    end
end)
