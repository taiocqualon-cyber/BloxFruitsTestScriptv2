-- 💀 DELTA GOD V8 FINAL – AI FRUIT HUNTER
repeat task.wait() until game:IsLoaded()

local player = game.Players.LocalPlayer
local Workspace = game:GetService("Workspace")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")
local TweenService = game:GetService("TweenService")

-- ===== HRP =====
local function getHRP()
	local char = player.Character or player.CharacterAdded:Wait()
	return char:WaitForChild("HumanoidRootPart")
end

-- ===== TARGET =====
local TargetFruits = {
	"kitsune","yeti","dough","t-rex","control",
	"tiger","mammoth","spirit","venom","gas",
	"gravity","portal","blizzard","pain"
}

local function isTarget(name)
	name = string.lower(name)
	for _,f in pairs(TargetFruits) do
		if string.find(name,f) then
			return true
		end
	end
	return false
end

-- ===== GUI =====
local gui = Instance.new("ScreenGui", game.CoreGui)
gui.ResetOnSpawn = false

local icon = Instance.new("TextButton", gui)
icon.Size = UDim2.new(0,40,0,40)
icon.Position = UDim2.new(0.02,0,0.15,0)
icon.Text = "👑"
icon.BackgroundColor3 = Color3.fromRGB(255,215,0)

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,260,0,150)
frame.Position = UDim2.new(0,-300,0.15,0)
frame.BackgroundColor3 = Color3.fromRGB(15,15,15)

local status = Instance.new("TextLabel", frame)
status.Size = UDim2.new(1,0,1,0)
status.TextColor3 = Color3.fromRGB(255,255,255)
status.BackgroundTransparency = 1
status.TextWrapped = true
status.Text = "👑 GOD V8 READY"

local open = false
icon.MouseButton1Click:Connect(function()
	open = not open
	TweenService:Create(frame, TweenInfo.new(0.4), {
		Position = open and UDim2.new(0.02,0,0.15,0) or UDim2.new(0,-300,0.15,0)
	}):Play()
end)

-- ===== ESP =====
local function ESP(obj)
	if obj:FindFirstChild("GOD") then return end
	
	local bill = Instance.new("BillboardGui", obj)
	bill.Name = "GOD"
	bill.Size = UDim2.new(0,100,0,40)
	bill.AlwaysOnTop = true
	
	local txt = Instance.new("TextLabel", bill)
	txt.Size = UDim2.new(1,0,1,0)
	txt.BackgroundTransparency = 1
	txt.TextColor3 = Color3.fromRGB(255,0,0)
	txt.TextStrokeTransparency = 0
	txt.Text = obj.Name
end

-- ===== FIND =====
local function findFruit()
	for _,v in pairs(Workspace:GetDescendants()) do
		if v:IsA("Tool") and v:FindFirstChild("Handle") then
			if string.find(string.lower(v.Name),"fruit") and isTarget(v.Name) then
				ESP(v.Handle)
				return v
			end
		end
	end
end

-- ===== EFFECT =====
local function effect()
	local hrp = getHRP()
	
	local p = Instance.new("ParticleEmitter", hrp)
	p.Rate = 200
	p.Lifetime = NumberRange.new(1)
	p.Speed = NumberRange.new(6)
	p.Size = NumberSequence.new(2)
	p.Color = ColorSequence.new(Color3.fromRGB(255,0,0))
	
	task.delay(1,function() p:Destroy() end)
end

-- ===== SMART SERVER =====
local visited = {}

local function getServers()
	local success, result = pcall(function()
		return game:HttpGet(
			"https://games.roblox.com/v1/games/"..
			game.PlaceId..
			"/servers/Public?limit=100"
		)
	end)

	if success then
		local data = HttpService:JSONDecode(result)
		return data.data or {}
	end
	return {}
end

local function smartHop()
	status.Text = "🧠 AI đang chọn server..."

	local servers = getServers()
	local best = {}

	for _,s in pairs(servers) do
		if s.playing < s.maxPlayers and not visited[s.id] then
			if s.playing <= 4 then -- 🔥 server fresh
				table.insert(best, s)
			end
		end
	end

	if #best == 0 then
		for _,s in pairs(servers) do
			if s.playing < s.maxPlayers and not visited[s.id] then
				table.insert(best, s)
			end
		end
	end

	if #best == 0 then
		visited = {}
		return
	end

	local pick = best[math.random(1,#best)]
	visited[pick.id] = true

	status.Text = "🚀 Hop..."

	task.wait(5) -- ⚠️ DELTA SAFE

	pcall(function()
		TeleportService:TeleportToPlaceInstance(game.PlaceId, pick.id, player)
	end)
end

-- ===== MAIN =====
while true do
	local fruit = findFruit()

	if fruit then
		status.Text = "💎 "..fruit.Name

		local hrp = getHRP()
		hrp.CFrame = fruit.Handle.CFrame + Vector3.new(0,3,0)

		local cd = fruit:FindFirstChildOfClass("ClickDetector")
		if cd then fireclickdetector(cd) end

		effect()
		task.wait(2)
	else
		smartHop()
	end

	task.wait(1)
end
