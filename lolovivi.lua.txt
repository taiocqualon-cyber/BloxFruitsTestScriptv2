-- Delta Super Delta x3 - Max Speed Legendary Fruit Collector
-- Nhặt trái mục tiêu, tránh trùng server, thống kê + popup

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local LocalPlayer = Players.LocalPlayer
local HttpService = game:GetService("HttpService")
local RunService = game:GetService("RunService")

-- Danh sách trái mục tiêu
local TargetFruits = {
    "kitsune","yeti","dough","t-rex","control",
    "tiger","mammoth","spirit","venom","gas",
    "gravity","portal","blizzard","pain"
}

-- Biến thống kê
local fruitStats = {}
for _, fruit in pairs(TargetFruits) do
    fruitStats[fruit] = 0
end

-- Kiểm tra trái có nằm trong danh sách
local function isTargetFruit(fruitName)
    for _, name in pairs(TargetFruits) do
        if fruitName:lower() == name:lower() then
            return true
        end
    end
    return false
end

-- Lưu server đã ghé
local visitedServers = {}

-- Hop server, tránh trùng
local function hopServerUnique()
    local placeId = game.PlaceId
    local servers = {}
    local success, response = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..placeId.."/servers/Public?sortOrder=Asc&limit=100")
    end)
    if success then
        local data = HttpService:JSONDecode(response)
        for _, server in pairs(data.data) do
            if server.playing < server.maxPlayers and not visitedServers[server.id] then
                table.insert(servers, server.id)
            end
        end
    end

    if #servers > 0 then
        local randomServer = servers[math.random(1, #servers)]
        visitedServers[randomServer] = true
        TeleportService:TeleportToPlaceInstance(placeId, randomServer, LocalPlayer)
    else
        print("Không còn server mới trống. Đợi 5s rồi thử lại...")
        wait(5)
        hopServerUnique()
    end
end

-- Nhặt trái mục tiêu cực nhanh, cập nhật thống kê
local function pickupTargetFruits()
    for _, fruit in pairs(workspace:GetDescendants()) do
        if fruit:IsA("Tool") and isTargetFruit(fruit.Name) then
            pcall(function()
                fruit.Handle.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame
                firetouchinterest(fruit.Handle, LocalPlayer.Character.HumanoidRootPart, 0)
                firetouchinterest(fruit.Handle, LocalPlayer.Character.HumanoidRootPart, 1)
            end)
            fruitStats[fruit.Name] = fruitStats[fruit.Name] + 1
            print("✅ Đã nhặt trái: "..fruit.Name.." | Tổng gặp: "..fruitStats[fruit.Name])
            -- Popup đơn giản trên màn hình
            pcall(function()
                local BillboardGui = Instance.new("BillboardGui")
                BillboardGui.Size = UDim2.new(0,200,0,50)
                BillboardGui.Adornee = LocalPlayer.Character.Head
                BillboardGui.AlwaysOnTop = true
                local TextLabel = Instance.new("TextLabel", BillboardGui)
                TextLabel.Size = UDim2.new(1,0,1,0)
                TextLabel.BackgroundTransparency = 1
                TextLabel.TextColor3 = Color3.new(1,1,0)
                TextLabel.TextStrokeTransparency = 0
                TextLabel.Text = "Nhặt được: "..fruit.Name
                BillboardGui.Parent = LocalPlayer.PlayerGui
                game.Debris:AddItem(BillboardGui,3) -- tự biến mất sau 3 giây
            end)
        end
    end
end

-- Vòng lặp siêu nhanh
while true do
    pickupTargetFruits()
    local backpackFruit = LocalPlayer.Backpack:FindFirstChildWhichIsA("Tool")
    if backpackFruit and isTargetFruit(backpackFruit.Name) then
        print("🎉 Đã có trái mục tiêu: "..backpackFruit.Name.." | Tổng lần nhặt: "..fruitStats[backpackFruit.Name])
        break
    else
        print("Chưa tìm thấy trái mục tiêu, hop server cực nhanh...")
        hopServerUnique()
        wait(0.3) -- Max tốc độ hop server Delta
    end
end
