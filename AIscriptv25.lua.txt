-- ===== HÀM TẠO TOGGLE (QUICK SELECT) =====
local function addHack(name)
    local btn = Instance.new("TextButton", container)
    btn.Size = UDim2.new(1, 0, 0, 45)
    btn.BackgroundColor3 = Color3.fromRGB(30, 0, 0)
    btn.Text = name .. " : OFF"
    btn.Font = Enum.Font.GothamBold
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.TextSize = 14
    Instance.new("UICorner", btn)

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.Text = name .. (active and " : ON" or " : OFF")
        btn.BackgroundColor3 = active and Color3.fromRGB(150, 0, 0) or Color3.fromRGB(30, 0, 0)
    end)
    return function() return active end
end

-- ===== LOGIC "HACK" THỰC THỤ (ATTACK ENGINE) =====

-- Hàm gây sát thương cực mạnh
local function rawAttack(target)
    if target and target:FindFirstChild("Humanoid") and target.Humanoid.Health > 0 then
        local remotes = ReplicatedStorage:WaitForChild("Remotes")
        -- Gửi liên tiếp 3 đòn đánh trong 1 lần gọi (Tripple Damage Simulation)
        for i = 1, 3 do
            remotes.CommF_:InvokeServer("AttackRemote", target.HumanoidRootPart.CFrame)
            remotes.Validator:FireServer(tick(), 0) -- Bypass validator cơ bản
        end
    end
end

-- Kill Aura & Mob Bring (Gom quái cực mạnh)
RunService.Stepped:Connect(function()
    if getKillAura() or getFastAttack() then
        for _, enemy in pairs(workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") then
                local dist = (enemy.HumanoidRootPart.Position - hrp.Position).Magnitude
                if dist < 65 then
                    -- Gom toàn bộ quái về trước mặt
                    enemy.HumanoidRootPart.CFrame = hrp.CFrame * CFrame.new(0, 0, -5)
                    enemy.HumanoidRootPart.CanCollide = false
                    rawAttack(enemy) -- Xả sát thương
                end
            end
        end
    end
end)

end)
