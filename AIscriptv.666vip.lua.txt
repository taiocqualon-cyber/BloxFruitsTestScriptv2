-- 1. Logic Farm & Gom Quái
function GhostFarmLogic()
    spawn(function()
        while _G.AutoFarm do
            task.wait(0.1)
            pcall(function()
                local p = game.Players.LocalPlayer
                local char = p.Character
                
                -- Tự động cầm Light
                local tool = p.Backpack:FindFirstChild(_G.Weapon) or char:FindFirstChild(_G.Weapon)
                if tool and tool.Parent ~= char then char.Humanoid:EquipTool(tool) end

                for _, v in pairs(game.Workspace.Enemies:GetChildren()) do
                    if (v.Name == "Snow Bandit" or v.Name == "Snowman") and v.Humanoid.Health > 0 then
                        char.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame * CFrame.new(0, 15, 0)
                        if _G.BringMob then
                            v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, -10, 0)
                            v.HumanoidRootPart.CanCollide = false
                        end
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                    end
                end
            end)
        end
    end)
end

-- 2. Logic Nhặt Rương
function GhostChestLogic()
    spawn(function()
        while _G.AutoChest do
            task.wait(0.1)
            for _, v in pairs(game.Workspace:GetChildren()) do
                if v.Name:find("Chest") and v:IsA("Part") then
                    local dist = (v.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                    local tween = game:GetService("TweenService"):Create(game.Players.LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(dist/_G.TweenSpeed), {CFrame = v.CFrame})
                    tween:Play()
                    tween.Completed:Wait()
                end
            end
        end
    end)
end

-- 3. Logic ESP Fruit (Hiện khoảng cách)
spawn(function()
    while task.wait(1) do
        if _G.FruitESP then
            for _, v in pairs(game.Workspace:GetChildren()) do
                if v:IsA("Tool") and v.Name:find("Fruit") and not v.Handle:FindFirstChild("GhostESP") then
                    local b = Instance.new("BillboardGui", v.Handle)
                    b.Name = "GhostESP"
                    b.AlwaysOnTop = true
                    b.Size = UDim2.new(0, 100, 0, 50)
                    local t = Instance.new("TextLabel", b)
                    t.Size = UDim2.new(1, 0, 1, 0)
                    t.BackgroundTransparency = 1
                    t.TextColor3 = Color3.new(1, 1, 0)
                    spawn(function()
                        while v:IsA("Tool") and v.Parent == game.Workspace do
                            t.Text = v.Name .. "\n[" .. math.floor((v.Handle.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude) .. "m]"
                            task.wait(0.5)
                        end
                    end)
                end
            end
        end
    end
end)

-- 4. Logic Hop Server
function SimpleHop()
    local Http = game:GetService("HttpService")
    local Api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
    local Raw = game:HttpGet(Api)
    local Servers = Http:JSONDecode(Raw)
    for _, s in pairs(Servers.data) do
        if s.playing < s.maxPlayers and s.id ~= game.JobId then
            game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, s.id, game.Players.LocalPlayer)
            break
        end
    end
end

function CheckMoonAndHop()
    if game:GetService("Lighting").ClockTime >= 18 or game:GetService("Lighting").ClockTime <= 5 then
        Rayfield:Notify({Title = "Moon Status", Content = "Đang đêm! Tìm server Full Moon...", Duration = 3})
    end
    SimpleHop()
end

function FruitHopCheck()
    local found = false
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and v.Name:find("Fruit") then found = true break end
    end
    if found then Rayfield:Notify({Title = "Fruit!", Content = "Có trái! Đừng hop!", Duration = 5})
    else SimpleHop() end
end
