-- ===== HÀM KIỂM TRA NPC ĐỘNG (DYNAMIC FINDER) =====
local function getNPCCFrame(npcName)
    local npcFolder = workspace:FindFirstChild("NPCs")
    if npcFolder then
        local npc = npcFolder:FindFirstChild(npcName)
        if npc and npc:IsA("Model") and npc.PrimaryPart then
            return npc.PrimaryPart.CFrame
        end
    end
    return nil
end

-- ===== HÀM TẤN CÔNG AN TOÀN (SAFE ATTACK) =====
local function safeAttack(target)
    if target and target:FindFirstChild("Humanoid") and target.Humanoid.Health > 0 then
        local remotes = ReplicatedStorage:FindFirstChild("Remotes")
        if remotes then
            local validator = remotes:FindFirstChild("Validator")
            local commF = remotes:FindFirstChild("CommF_")
            
            if validator then validator:FireServer(tick(), 0) end
            if commF then commF:InvokeServer("AttackRemote", target.HumanoidRootPart.CFrame) end
        end
        
        -- Kiểm tra Module Net Attack
        local netModule = ReplicatedStorage:FindFirstChild("Modules") and ReplicatedStorage.Modules:FindFirstChild("Net")
        if netModule and netModule:FindFirstChild("Attack") then
            netModule.Attack:FireServer()
        end
    end
end

-- ===== LOGIC KIỂM TRA QUEST GUI =====
local function isQuestActive()
    local playerGui = player:FindFirstChild("PlayerGui")
    if playerGui then
        local mainGui = playerGui:FindFirstChild("Main")
        if mainGui and mainGui:FindFirstChild("Quest") then
            return mainGui.Quest.Visible
        end
    end
    return false
end

-- ===== HÀM DI CHUYỂN TWEEN =====
local function safeTween(targetCF)
    local dist = (hrp.Position - targetCF.p).Magnitude
    local tween = TweenService:Create(hrp, TweenInfo.new(dist/150, Enum.EasingStyle.Linear), {CFrame = targetCF})
    tween:Play()
    return tween
end

-- ===== CHƯƠNG TRÌNH ĐIỀU KHIỂN CHÍNH (LOOP) =====
task.spawn(function()
    while task.wait(0.2) do -- Tăng lên 0.2s để giảm tải cho CPU (Performance)
        if isAuto then
            local myLevel = player.Data.Level.Value
            local cq = nil
            
            for i = #Sea1_Quests, 1, -1 do
                if myLevel >= Sea1_Quests[i].Level then cq = Sea1_Quests[i]; break end
            end
            
            if cq then
                if not isQuestActive() then
                    -- Tìm NPC động, nếu không thấy dùng tọa độ tĩnh dự phòng
                    local npcCF = getNPCCFrame(cq.NPC) or CFrame.new(cq.Pos)
                    safeTween(npcCF * CFrame.new(0, 5, 0))
                    task.wait(0.7)
                    local commF = ReplicatedStorage:FindFirstChild("Remotes") and ReplicatedStorage.Remotes:FindFirstChild("CommF_")
                    if commF then commF:InvokeServer("StartQuest", cq.QuestName, 1) end
                else
                    local enemyFound = false
                    for _, e in pairs(workspace.Enemies:GetChildren()) do
                        if e.Name == cq.Monster and e:FindFirstChild("Humanoid") and e.Humanoid.Health > 0 then
                            enemyFound = true
                            -- Tween lơ lửng trên đầu
                            hrp.CFrame = e.HumanoidRootPart.CFrame * CFrame.new(0, 20, 0)
                            -- Gom quái (Mob Bring)
                            e.HumanoidRootPart.CFrame = hrp.CFrame * CFrame.new(0, -20, 0)
                            e.HumanoidRootPart.CanCollide = false
                            -- Tấn công
                            safeAttack(e)
                        end
                    end
                    if not enemyFound then safeTween(CFrame.new(cq.Pos + Vector3.new(0, 60, 0))) end
                end
            end
        end
    end
end)
