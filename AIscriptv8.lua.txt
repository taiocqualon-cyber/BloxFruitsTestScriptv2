repeat wait() until game:IsLoaded()

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")

local plr = Players.LocalPlayer

-- ===== GUI =====
local gui = Instance.new("ScreenGui", plr.PlayerGui)

local icon = Instance.new("TextButton", gui)
icon.Size = UDim2.new(0,40,0,40)
icon.Position = UDim2.new(0,10,0.5,0)
icon.Text = "👹"

local status = Instance.new("TextLabel", gui)
status.Size = UDim2.new(0,220,0,40)
status.Position = UDim2.new(0,60,0.5,0)
status.BackgroundColor3 = Color3.fromRGB(0,0,0)
status.TextColor3 = Color3.fromRGB(0,255,0)
status.TextScaled = true
status.Text = "Đang khởi động..."

-- ===== CONFIG =====
local checkedServers = {}

local goodFruits = {
    ["Kitsune"]=true,
    ["Dragon"]=true,
    ["Dough"]=true,
    ["Leopard"]=true
}

-- ===== FUNCTION =====
local function setStatus(text)
    status.Text = text
    warn(text)
end

local function isFruit(v)
    return v:IsA("Tool") and v:FindFirstChild("Handle") and v.Parent == workspace
end

local function isGood(v)
    for name,_ in pairs(goodFruits) do
        if string.find(v.Name,name) then return true end
    end
    return false
end

local function getServers()
    local list = {}

    local ok,res = pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
    end)

    if ok then
        local data = HttpService:JSONDecode(res)
        for _,s in pairs(data.data) do
            if s.playing < s.maxPlayers-2 and not checkedServers[s.id] then
                table.insert(list,s.id)
            end
        end
    end

    return list
end

local function hop()
    setStatus("🔁 Đang đổi server...")

    local servers = getServers()

    if #servers > 0 then
        local id = servers[math.random(1,#servers)]
        checkedServers[id] = true

        TeleportService:TeleportToPlaceInstance(
            game.PlaceId,
            id,
            plr
        )
    else
        setStatus("💀 Reset danh sách server")
        checkedServers = {}
    end
end

-- ===== LOOP =====
while true do
    wait(3)

    setStatus("🔍 Đang scan server...")

    local foundAny = false
    local foundGood = false

    for _,v in pairs(workspace:GetDescendants()) do
        if isFruit(v) then
            foundAny = true

            if isGood(v) then
                foundGood = true

                setStatus("🔥 TÌM THẤY: "..v.Name)

                local char = plr.Character
                if char and char:FindFirstChild("HumanoidRootPart") then
                    char.HumanoidRootPart.CFrame =
                        v.Handle.CFrame + Vector3.new(0,3,0)
                end

                break
            end
        end
    end

    if not foundAny then
        setStatus("❌ Không có fruit")
        hop()

    elseif not foundGood then
        setStatus("🍈 Có fruit thường → đổi")
        hop()

    else
        setStatus("👹 Đã giữ server")
        break
    end
end
