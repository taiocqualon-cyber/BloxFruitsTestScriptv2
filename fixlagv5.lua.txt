    - REDUCTION: 70% Skill Effects (Keep 30% for Hitbox).

if not game:IsLoaded() then game.Loaded:Wait() end

local Lighting = game:GetService("Lighting")
local RunService = game:GetService("RunService")

-- 1. UNLOCK & LOCK 60 FPS (KHÔNG CHO TỤT)
if setfpscap then
    setfpscap(60)
end

-- 2. HÀM FIX CỰC MƯỢT (CLEANER)
local function UltimateFix()
    -- Hạ cấp Rendering xuống mức đáy
    settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
    
    -- XÓA BẦU TRỜI (BLACK WORLD)
    for _, v in pairs(Lighting:GetChildren()) do
        if v:IsA("Sky") or v:IsA("Atmosphere") or v:IsA("Clouds") then
            v:Destroy()
        end
    end
    Lighting.GlobalShadows = false
    Lighting.FogEnd = 9e9
    Lighting.Brightness = 2 -- Tăng sáng để bù cho bầu trời đen
    Lighting.ClockTime = 14 -- Cố định giờ trưa cho sáng

    -- QUÉT VÀ CẮT TỈA
    for _, v in pairs(game:GetDescendants()) do
        -- 1. Xóa đổ bóng, texture nhà, nước, cây
        if v:IsA("Part") or v:IsA("MeshPart") or v:IsA("BasePart") then
            v.Material = Enum.Material.SmoothPlastic
            v.Reflectance = 0
            v.CastShadow = false
            -- Xóa vân nước/cỏ
            if v.Name == "Water" or v.Name == "Sea" or v.Name:find("Leaf") or v.Name:find("Grass") then
                v.Transparency = 1
                v.CanCollide = false
            end
        elseif v:IsA("Decal") or v:IsA("Texture") then
            v:Destroy()
            
        -- 2. CẮT HIỆU ỨNG CHIÊU (70% REDUCTION)
        elseif v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then
            -- Giữ lại 30% để biết đối thủ ra chiêu
            if v:IsA("ParticleEmitter") then
                v.Rate = v.Rate * 0.3 -- Cắt 70%, chỉ để lại 30% hạt
                v.LightEmission = 0.5
            elseif v:IsA("Trail") or v:IsA("Beam") then
                v.Lifetime = v.Lifetime * 0.3 -- Làm hiệu ứng biến mất nhanh hơn 70%
                v.Transparency = NumberSequence.new(0.5) -- Làm mờ bớt
            end
        end
    end
end

-- 3. GIẢI PHÓNG RAM LIÊN TỤC
task.spawn(function()
    while task.wait(5) do
        if collectgarbage then
            collectgarbage("collect")
        end
    end
end)

-- THỰC THI
pcall(UltimateFix)
