-- 1. Thuật toán Hop Server ngẫu nhiên (Tránh server cũ)
function AdvancedHop()
    local Http = game:GetService("HttpService")
    local Api = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Desc&limit=100"
    local Raw = game:HttpGet(Api)
    local Servers = Http:JSONDecode(Raw).data
    local Server = Servers[math.random(1, #Servers)]
    if Server.playing < Server.maxPlayers and Server.id ~= game.JobId then
        game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, Server.id, game.Players.LocalPlayer)
    else
        AdvancedHop()
    end
end

-- 2. Siêu Săn Fruit (Vào là quét)
function SuperFruitHop()
    local found = false
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and (v.Name:find("Fruit") or v:FindFirstChild("Handle")) then
            found = true break
        end
    end
    if found then
        Rayfield:Notify({Title = "PHÁT HIỆN!", Content = "Có trái! Đừng nhảy nữa!", Duration = 10})
        _G.FruitESP = true
    else
        AdvancedHop()
    end
end

-- 3. Logic Tween Di Chuyển (An Toàn)
function TweenToTarget(targetName)
    local targetObj = nil
    for _, v in pairs(game.Workspace:GetDescendants()) do
        if v.Name:find(targetName) and (v:IsA("Model") or v:IsA("Part")) then
            targetObj = v
            break
        end
    end
    
    if targetObj then
        local pos = targetObj:IsA("Model") and targetObj.PrimaryPart.CFrame or targetObj.CFrame
        local dist = (pos.p - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
        local tween = game:GetService("TweenService"):Create(game.Players.LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(dist/_G.TweenSpeed), {CFrame = pos})
        tween:Play()
    else
        Rayfield:Notify({Title = "Lỗi", Content = "Không tìm thấy mục tiêu trong server này!", Duration = 3})
    end
end

-- 4. Instant Teleport Fruit
function InstantTeleportFruit()
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") and v:FindFirstChild("Handle") then
            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.Handle.CFrame
            return
        end
    end
end

-- 5. ESP System (Fruit & NPC)
spawn(function()
    while task.wait(1.5) do
        if _G.FruitESP then
            for _, v in pairs(game.Workspace:GetChildren()) do

                    local b = Instance.new("BillboardGui", v.Handle); b.Name = "GhostESP"; b.AlwaysOnTop = true; b.Size = UDim2.new(0, 100, 0, 50)
                    local t = Instance.new("TextLabel", b); t.Size = UDim2.new(1,0,1,0); t.BackgroundTransparency = 1; t.TextColor3 = Color3.new(1,1,0)
                    spawn(function() while v.Parent == game.Workspace do t.Text = v.Name .. "\n[" .. math.floor((v.Handle.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude) .. "m]"; task.wait(0.5) end end)
                end
            end
        end
        if _G.NPC_ESP then
            for _, v in pairs(game.Workspace:GetDescendants()) do
                if (v.Name:find("Dealer") or v.Name:find("Sword") or v.Name:find("Color")) and v:IsA("Model") and not v:FindFirstChild("GhostNPC") then
                    local b = Instance.new("BillboardGui", v); b.Name = "GhostNPC"; b.AlwaysOnTop = true; b.Size = UDim2.new(0, 150, 0, 50)
                    local t = Instance.new("TextLabel", b); t.Size = UDim2.new(1,0,1,0); t.BackgroundTransparency = 1; t.TextColor3 = Color3.new(0,1,1)
                    t.Text = "NPC: " .. v.Name
                end
            end
        end
    end
end)

-- 6. Check Moon Phase
function CheckMoonAndHop()
    if game:GetService("Lighting").ClockTime >= 18 or game:GetService("Lighting").ClockTime <= 5 then
        Rayfield:Notify({Title = "Moon Status", Content = "Đang ban đêm!", Duration = 3})
    end
    AdvancedHop()
end
