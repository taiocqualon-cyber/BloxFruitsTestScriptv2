-- 👹 FINAL POLISH HUB

local Players = game:GetService("Players")
local plr = Players.LocalPlayer
local TweenService = game:GetService("TweenService")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

-- ===== UI =====
local gui = Instance.new("ScreenGui", plr.PlayerGui)

local icon = Instance.new("TextButton", gui)
icon.Size = UDim2.new(0,50,0,50)
icon.Position = UDim2.new(0,10,0.5,0)
icon.Text = "👹"
icon.BackgroundColor3 = Color3.fromRGB(0,0,0)
icon.TextColor3 = Color3.fromRGB(255,0,0)

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0,300,0,350)
frame.Position = UDim2.new(0.5,-150,0.5,-175)
frame.BackgroundColor3 = Color3.fromRGB(15,0,0)
frame.Visible = false
frame.Active = true
frame.Draggable = true

icon.MouseButton1Click:Connect(function()
    frame.Visible = not frame.Visible
end)

-- ===== NOTIFY =====
local function notify(msg)
    print("💀 "..msg)
end

-- ===== TOGGLE (MƯỢT) =====
local function toggleBtn(name,y,callback)
    local state=false

    local btn=Instance.new("TextButton",frame)
    btn.Size=UDim2.new(1,0,0,40)
    btn.Position=UDim2.new(0,0,0,y)
    btn.BackgroundColor3=Color3.fromRGB(30,0,0)
    btn.Text=name.." : OFF"
    btn.TextColor3=Color3.fromRGB(255,0,0)

    btn.MouseButton1Click:Connect(function()
        state=not state

        TweenService:Create(btn,TweenInfo.new(0.2),{
            BackgroundColor3 = state and Color3.fromRGB(0,100,0) or Color3.fromRGB(30,0,0)
        }):Play()

        btn.Text=name.." : "..(state and "ON" or "OFF")
        callback(state)
    end)
end

-- ===== STATES =====
local autoChest,espFruit,espNPC = false,false,false

-- ===== ESP DISTANCE =====
local function addESP(v,text)
    if v:FindFirstChild("ESP") then return end

    local b=Instance.new("BillboardGui",v)
    b.Name="ESP"
    b.Size=UDim2.new(0,120,0,40)
    b.AlwaysOnTop=true

    local t=Instance.new("TextLabel",b)
    t.Size=UDim2.new(1,0,1,0)
    t.BackgroundTransparency=1
    t.TextColor3=Color3.fromRGB(255,255,0)

    game:GetService("RunService").RenderStepped:Connect(function()
        if v and v.Parent and plr.Character then
            local dist=(plr.Character.HumanoidRootPart.Position - v.Position).Magnitude
            t.Text=text.." ["..math.floor(dist).."]"
        end
    end)
end

-- ===== FRUIT CHECK =====
local function isFruit(v)
    return v:IsA("Tool") and v:FindFirstChild("Handle") and v.Parent==workspace
end

-- ===== NPC =====
local function isNPC(v)
    return v:IsA("Model") and v.Name=="Master of Auras"
end

-- ===== STORE =====
local function store()
    for _,t in pairs(plr.Backpack:GetChildren()) do
        if t:IsA("Tool") then
            pcall(function()
                ReplicatedStorage.Remotes.CommF_:InvokeServer("StoreFruit",t.Name)
            end)
        end
    end
end

-- ===== TWEEN =====
local function tweenTo(pos)
    local hrp=plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local dist=(hrp.Position-pos).Magnitude

    local tw=TweenService:Create(hrp,
        TweenInfo.new(dist/350,Enum.EasingStyle.Linear),
        {CFrame=CFrame.new(pos)}
    )
    tw:Play()
    tw.Completed:Wait()
end

-- ===== CHEST =====
local function getChests()
    local c={}
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and string.find(v.Name:lower(),"chest") then
            if v:FindFirstChild("HumanoidRootPart") then
                table.insert(c,v.HumanoidRootPart)
            end
        end
    end
    return c
end

-- ===== SERVER =====
local visited={}
local function changeServer()
    notify("🔁 Changing server...")

    local list={}
    local ok,res=pcall(function()
        return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?limit=100")
    end)

    if ok then
        local data=HttpService:JSONDecode(res)
        for _,s in pairs(data.data) do
            if s.playing < s.maxPlayers-2 and not visited[s.id] then
                table.insert(list,s.id)
            end
        end
    end

    if #list>0 then
        local id=list[math.random(1,#list)]
        visited[id]=true
        TeleportService:TeleportToPlaceInstance(game.PlaceId,id,plr)
    end
end

-- ===== BUTTONS =====
toggleBtn("Auto Chest",10,function(v) autoChest=v end)
toggleBtn("ESP Fruits",60,function(v) espFruit=v end)
toggleBtn("ESP NPC",110,function(v) espNPC=v end)

-- TELEPORT
local tp=Instance.new("TextButton",frame)
tp.Size=UDim2.new(1,0,0,40)
tp.Position=UDim2.new(0,0,0,170)
tp.Text="🚀 Teleport Fruit"

tp.MouseButton1Click:Connect(function()
    for _,v in pairs(workspace:GetDescendants()) do
        if isFruit(v) then
            plr.Character.HumanoidRootPart.CFrame=v.Handle.CFrame
            break
        end
    end
end)

-- SERVER BTN
local sv=Instance.new("TextButton",frame)
sv.Size=UDim2.new(1,0,0,40)
sv.Position=UDim2.new(0,0,0,220)
sv.Text="🔁 Change Server"
sv.MouseButton1Click:Connect(changeServer)

-- ===== LOOP =====
while true do

    for _,v in pairs(workspace:GetDescendants()) do
        if espFruit and isFruit(v) then
            addESP(v,"🍈")
        end

        if espNPC and isNPC(v) and v:FindFirstChild("HumanoidRootPart") then
            addESP(v.HumanoidRootPart,"👹 HAKI")
        end
    end

    if autoChest then
        local chests=getChests()

        for _,c in pairs(chests) do
            tweenTo(c.Position + Vector3.new(0,3,0))
            notify("💰 Chest")
            wait(0.2)
        end

        if #chests==0 then
            changeServer()
        end
    end

    store()

    wait(1)
end
