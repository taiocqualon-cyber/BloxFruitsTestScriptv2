-- [[ BIẾN HỆ THỐNG ]] --
_G.AutoFarm = false

   Callback = function(Value)
      _G.AttackDelay = Value
   end,
})

-- [[ HÀM XỬ LÝ FARM TRONG BÓNG TỐI ]] --
function GhostFarmLogic()
    spawn(function()
        while _G.AutoFarm do
            task.wait(0.1)
            pcall(function()
                local player = game.Players.LocalPlayer
                local char = player.Character
                
                -- Tự động cầm vũ khí (Light Fruit)
                local fruit = player.Backpack:FindFirstChild(_G.SelectedWeapon) or char:FindFirstChild(_G.SelectedWeapon)
                if fruit and fruit.Parent ~= char then
                    char.Humanoid:EquipTool(fruit)
                end

                -- Tìm quái mục tiêu (Tối ưu cho đảo Tuyết - Snowman/Snow Bandit)
                for _, mob in pairs(game.Workspace.Enemies:GetChildren()) do
                    if (mob.Name == "Snowman" or mob.Name == "Snow Bandit") and mob:FindFirstChild("Humanoid") and mob.Humanoid.Health > 0 then
                        
                        -- Di chuyển mượt mà tới quái
                        char.HumanoidRootPart.CFrame = mob.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                        
                        -- Gom quái Bring Mob
                        mob.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                        mob.HumanoidRootPart.CanCollide = false
                        
                        -- Tấn công gây Dame thực (Đã fix lỗi ảo)
                        game:GetService("VirtualUser"):CaptureController()
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                        task.wait(_G.AttackDelay)
                    end
                end
            end)
        end
    end)
end
