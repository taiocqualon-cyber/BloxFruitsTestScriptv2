local Window = Fluent:CreateWindow({
    Title = "Pháp Sư Hub | Blox Fruits",
    SubTitle = "v1.0 - By YourName",
    TabWidth = 160,
    Size = UDim2.fromOffset(580, 460),
    Acrylic = true,
    Theme = "Dark"

local Tabs = {
    Main = Window:AddTab({ Title = "Auto Farm", Icon = "rbxassetid://4483345998" }),
    Visuals = Window:AddTab({ Title = "Tiện ích", Icon = "rbxassetid://4483345998" })
}

-- CHỨC NĂNG 1: AUTO NHẶT RƯƠNG (Chest Farm)
Tabs.Main:AddToggle("AutoChest", {Title = "Auto Nhặt Rương (Safe)", Default = false }):OnChanged(function(Value)
    _G.AutoChest = Value
    spawn(function()
        while _G.AutoChest do
            task.wait(0.1)
            for _, v in pairs(game:GetService("Workspace"):GetChildren()) do
                if v:IsA("Model") and (v.Name == "Chest1" or v.Name == "Chest2" or v.Name == "Chest3") then
                    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.PrimaryPart.CFrame
                    task.wait(0.5) -- Tránh bị Kick vì di chuyển quá nhanh
                end
            end
        end
    end)
end)

-- CHỨC NĂNG 2: AUTO FARM LEVEL (Khung cơ bản)
Tabs.Main:AddToggle("AutoFarm", {Title = "Auto Farm Level (Beta)", Default = false }):OnChanged(function(Value)
    _G.AutoFarm = Value
    -- Logic Auto Farm ở đây cực kỳ phức tạp (Check Level -> Nhận Quest -> Di chuyển -> Đánh)
    -- Lời khuyên: Dùng lệnh gọi các Hub lớn bên dưới để tối ưu
    if Value then
        Fluent:Notify({ Title = "Thông báo", Content = "Đang khởi động hệ thống Farm..." })
    end
end)

-- CHỨC NĂNG 3: HIỂN THỊ TRÁI ÁC QUỶ (Fruit ESP)

                Fluent:Notify({ Title = "Tìm thấy!", Content = "Trái: " .. v.Name })
                -- Thêm logic tạo Highlight ở đây
            end
        end
    end
})
