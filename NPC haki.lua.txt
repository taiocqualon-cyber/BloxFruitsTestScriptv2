-- ================= CHECK NPC =================
local function isHakiNPC(v)
	if v:IsA("Model") and v:FindFirstChild("Humanoid") then
		local name = v.Name:lower()
		return name:find("legendary") or name:find("color")
	end
end

-- ================= ESP =================
local function createESP(npc)
	if npc:FindFirstChild("HakiESP") then return end
	
	local head = npc:FindFirstChild("Head")
	if not head then return end
	
	local esp = Instance.new("BillboardGui", npc)
	esp.Name = "HakiESP"
	esp.Size = UDim2.new(0,150,0,40)
	esp.Adornee = head
	esp.AlwaysOnTop = true

	local text = Instance.new("TextLabel", esp)
	text.Size = UDim2.new(1,0,1,0)
	text.BackgroundTransparency = 1
	text.TextScaled = true
	text.TextColor3 = Color3.fromRGB(255,255,0)

	task.spawn(function()
		while npc.Parent do
			task.wait(0.3)
			local char = player.Character
			if char and char:FindFirstChild("HumanoidRootPart") then
				local dist = (head.Position - char.HumanoidRootPart.Position).Magnitude
				text.Text = "⭐ HAKI LEGEND - "..math.floor(dist).."m"
			end
		end
	end)
end
