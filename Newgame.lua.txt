-- 1. TẠO SÀN ĐẤU (ARENA)
local floor = Instance.new("Part")
floor.Name = "GhostArena"
floor.Size = Vector3.new(100, 1, 100)
floor.CFrame = game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame * CFrame.new(0, -5, 0)
floor.Anchored = true
floor.BrickColor = BrickColor.new("Dark stone grey")
floor.Material = Enum.Material.Concrete
floor.Parent = game.Workspace

-- 2. TẠO QUÁI ZOMBIE (MỤC TIÊU)
local function TaoZombie(pos)
    local zombie = Instance.new("Model", game.Workspace)
    zombie.Name = "Zombie_ThaiHoang"
    local dummy = Instance.new("Part", zombie)
    dummy.Name = "Head"
    dummy.Size = Vector3.new(2,2,2)
    dummy.CFrame = pos
    local hum = Instance.new("Humanoid", zombie)
    hum.MaxHealth = 100
    hum.Health = 100
    print("Đã triệu hồi Zombie!")
end

-- Triệu hồi 3 con Zombie xung quanh
for i = 1, 3 do
    TaoZombie(floor.CFrame * CFrame.new(math.random(-20,20), 5, math.random(-20,20)))
end

-- 3. TẶNG KIẾM SKILL (CHO NGÀI QUẨY)
local tool = Instance.new("Tool")
tool.Name = "Thanh Kiem Ghost"
tool.RequiresHandle = true
tool.Parent = game.Players.LocalPlayer.StarterPack -- Hoặc BackPack

local handle = Instance.new("Part", tool)
handle.Name = "Handle"
handle.Size = Vector3.new(1, 5, 1)
handle.BrickColor = BrickColor.new("Neon orange")

-- Script Kỹ năng cho Kiếm
tool.Activated:Connect(function()
    local char = game.Players.LocalPlayer.Character
    local fx = Instance.new("Part", game.Workspace)
    fx.Shape = Enum.PartType.Ball
    fx.Size = Vector3.new(10, 10, 10)
    fx.CFrame = char.HumanoidRootPart.CFrame
    fx.CanCollide = false
    fx.Transparency = 0.5
    fx.BrickColor = BrickColor.new("Really red")
    
    -- Gây sát thương xung quanh (AOE)
    for _, v in pairs(game.Workspace:GetDescendants()) do
        if v:IsA("Humanoid") and v.Parent.Name ~= game.Players.LocalPlayer.Name then
            v:TakeDamage(50)
        end
    end
    
    wait(0.2)
    fx:Destroy()
end)

game.StarterGui:SetCore("SendNotification", {
    Title = "GHOST STUDIO",
    Text = "Game đã sẵn sàng! Kiểm tra túi đồ của ngài.",
    Duration = 5
})
