-- ===== CHECK MOON =====
local function GetMoonPhase()
	local clock = Lighting.ClockTime

	if clock >= 0 and clock < 4 then
		return 4, "🌕 Full Moon"
	elseif clock >= 4 and clock < 8 then
		return 3, "🌖 Sắp Full Moon"
	elseif clock >= 8 and clock < 16 then
		return 2, "🌗 Còn lâu"
	else
		return 1, "🌘 Còn lâu"
	end
end

-- ===== TIME TO FULL MOON =====
local function TimeToFullMoon()
	local clock = Lighting.ClockTime
	local timeLeft = (24 - clock) % 24
	return math.floor(timeLeft*60) .. "s"
end

-- ===== SERVER HOP =====
local function HopServer()
	local servers = {}
	local req = game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Asc&limit=100")
	local data = HttpService:JSONDecode(req)

	for _,v in pairs(data.data) do
		if v.playing < v.maxPlayers then
			table.insert(servers, v.id)
		end
	end

	if #servers > 0 then
		TeleportService:TeleportToPlaceInstance(game.PlaceId, servers[math.random(1,#servers)], player)
	end
end

-- ===== MAIN LOOP =====
local Active = false

toggle.MouseButton1Click:Connect(function()
	Active = not Active
	toggle.Text = Active and "ON" or "OFF"

	if Active then
		task.spawn(function()
			while Active do
				local level, text = GetMoonPhase()
				local timeLeft = TimeToFullMoon()

				status.Text = "Moon: "..text.."\nNext Full: "..timeLeft

				if level >= 3 then
					-- 🌖 hoặc 🌕 thì giữ server
					status.Text = status.Text .. "\n✅ Server tốt!"
				else
					-- 🌘 🌗 thì hop
					status.Text = status.Text .. "\n🔄 Đang đổi server..."
					HopServer()
				end

				task.wait(5)
			end
		end)
	end
end)
