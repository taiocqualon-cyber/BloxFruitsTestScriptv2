-- [[ LOGIC QUÉT TRỨNG & ESP ]] --
function TweenToTarget(name)
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v.Name:find("Egg") or v.Name:find("Easter") then
            local dist = (v.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
            game:GetService("TweenService"):Create(game.Players.LocalPlayer.Character.HumanoidRootPart, TweenInfo.new(dist/_G.TweenSpeed), {CFrame = v.CFrame}):Play()
            break
        end
    end
end

spawn(function()
    while task.wait(1.5) do
        if _G.EggESP then
            for _, v in pairs(game.Workspace:GetChildren()) do
                -- Quét các Object có tên liên quan đến Trứng sự kiện
                if (v.Name:find("Egg") or v.Name:find("Easter")) and not v:FindFirstChild("GhostEgg") then
                    local b = Instance.new("BillboardGui", v); b.Name = "GhostEgg"; b.AlwaysOnTop = true; b.Size = UDim2.new(0,100,0,50)
                    local t = Instance.new("TextLabel", b); t.Size = UDim2.new(1,0,1,0); t.BackgroundTransparency = 1; t.TextColor3 = Color3.fromRGB(255, 105, 180) -- Màu hồng Easter
                    t.Text = "🥚 EGG HERE!"
                end
            end
        end
    end
end)

function SecureHop()
    local Servers = game:GetService("HttpService"):JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Asc&limit=100")).data
    local s = Servers[math.random(1, #Servers)]
    game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, s.id, game.Players.LocalPlayer)
end
