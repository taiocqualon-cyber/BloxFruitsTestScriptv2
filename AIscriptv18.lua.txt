repeat wait() until game:IsLoaded()

local Players = game:GetService("Players")
local plr = Players.LocalPlayer

-- ===== STATE =====
local running = false
local RADIUS = 10000
local collected = {}

-- ===== DANH SÁCH ĐẢO =====
local ISLANDS = {
    Vector3.new(-385, 73, 300),      -- Cafe
    Vector3.new(-2600, 73, -5000),   -- Green Zone
    Vector3.new(1200, 150, -1500),   -- Snow Mountain
    Vector3.new(-5500, 100, -2500),  -- Hot & Cold
    Vector3.new(-300, 100, -1000)    -- Factory
}

local currentIsland = 1

-- ===== GUI =====
local gui = Instance.new("ScreenGui", plr.PlayerGui)

local btn = Instance.new("TextButton", gui)
btn.Size = UDim2.new(0,120,0,40)
btn.Position = UDim2.new(0,10,0.5,0)
btn.BackgroundColor3 = Color3.fromRGB(0,0,0)
btn.TextColor3 = Color3.fromRGB(0,255,0)
btn.TextScaled = true
btn.Text = "START"

btn.MouseButton1Click:Connect(function()
    running = not running
    btn.Text = running and "STOP" or "START"
end)

-- ===== HRP =====
local function getHRP()
    local char = plr.Character or plr.CharacterAdded:Wait()
    return char:WaitForChild("HumanoidRootPart")
end

-- ===== LẤY RƯƠNG GẦN NHẤT =====
local function getNearestChest()
    local hrp = getHRP()
    local nearest = nil
    local shortest = math.huge

    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and string.find(v.Name,"Chest") then

            if collected[v] then continue end

            for _,p in pairs(v:GetDescendants()) do
                if p:IsA("BasePart") then

                    local dist = (hrp.Position - p.Position).Magnitude

                    if dist <= RADIUS and dist < shortest then
                        shortest = dist
                        nearest = {model = v, pos = p.Position}
                    end

                    break
                end
            end

        end
    end

    return nearest
end

-- ===== TELEPORT =====
local function teleportTo(pos)
    local hrp = getHRP()
    hrp.CFrame = CFrame.new(pos + Vector3.new(0,3,0))
end

-- ===== ĐỔI ĐẢO =====
local function nextIsland()
    currentIsland = currentIsland + 1
    if currentIsland > #ISLANDS then
        currentIsland = 1
    end

    print("🌍 Sang đảo "..currentIsland)

    teleportTo(ISLANDS[currentIsland])
    wait(1)

    collected = {} -- reset danh sách rương
end

-- ===== MAIN =====
while true do
    wait(0.5)

    if running then
        local chest = getNearestChest()

        if chest then
            print("💰 Nhặt rương gần nhất")

            teleportTo(chest.pos)

            collected[chest.model] = true

            wait(0.5)
        else
            print("❌ Hết rương → đổi đảo")
            nextIsland()
        end
    end
end
